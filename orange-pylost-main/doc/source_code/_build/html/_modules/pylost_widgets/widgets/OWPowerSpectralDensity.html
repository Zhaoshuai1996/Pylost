[docs]    def update_csp(self):
        for item, dv in self.dataViewers.items():
            if dv.data() is not None:
                plt1 = dv.findChild(Plot1D)
                if plt1 is not None:
                    curves = plt1.getAllCurves()
                    for curve in curves:
                        if curve.getLegend()=='CSP':
                            continue
                        x, y, xerr, yerr = curve.getData()
                        q = np.isfinite(x)
                        if self.xmin!=0:
                            q = np.multiply(q, x>=self.xmin)
                        if self.xmax!=0:
                            q = np.multiply(q, x<=self.xmax)
                        xq = x[q]
                        yq = y[q]
                        if self.flip_csp:
                            yq = np.flip(yq)
                        yc = cumtrapz(yq, x=xq, initial=0)
                        if self.flip_csp:
                            yc = np.flip(yc)
                        plt1.addCurve(xq, yc, legend='CSP', linestyle='--', resetzoom=True,
                                      xlabel=curve.getXLabel(),
                                      ylabel='Cumulative Spectral Power',
                                      yaxis='right',
                                      copy=False)
                        break

[docs]    def apply_scan_item(self, Z, comment='', item=None):
        from scipy.fft import fft, fft2, fftfreq

        dims = super().get_detector_dimensions(Z)
        axes = dims.nonzero()[0]
        if Z.ndim > 1 and self.option == self.ALL_2D:
            if not any(axes):
                axes = [-2, -1]
            N = [Z.shape[axes[-2]], Z.shape[axes[-1]]]
            T = self.pix_size[-2:]
            unit_f = ['', '']
            if isinstance(Z, MetrologyData):
                arr = Z.get_axis_val_items_detector()[-2:]
                for i in range(2):
                    if np.any(arr[i].value):
                        T[i] = np.nanmean(np.diff(arr[i]))
                        unit_f[i] = '1/{}'.format(arr[i].unit)

            w1 = self.get_window(Z.ndim, N[-1], axes[-1])
            w2 = self.get_window(Z.ndim, N[-2], axes[-2])

            zf = fft2(Z*w1*w2, axes=axes[-2:])
            if isinstance(Z, MetrologyData):
                zf *= Z.unit
            slc = [slice(None)]*Z.ndim
            slc[axes[-1]] = slice(0,N[-1]//2)
            slc[axes[-2]] = slice(0,N[-2]//2)
            Zret = (2.0*T[-2]/N[-2])*(2.0*T[-1]/N[-1])*np.abs(zf[slc])**2
            slc[axes[-1]] = 0
            slc[axes[-2]] = 0
            Zret[slc] = Zret[slc]/2
            xf = [fftfreq(N[-2], T[-2])[:N[-2]//2], fftfreq(N[-1], T[-1])[:N[-1]//2]]

            if isinstance(Z, MetrologyData):
                pix_size = Z.pix_size_detector
                if not any(pix_size):
                    Z._set_dim_detector([True])
                pix_size = Z.pix_size_detector
                pix_size[-1] = Quantity(np.nanmean(np.diff(xf[-1])), unit=unit_f[-1])
                pix_size[-2] = Quantity(np.nanmean(np.diff(xf[-2])), unit=unit_f[-2])
                Zret = Z.copy_to(Zret)
                Zret._set_pix_size(pix_size)
                Zret._set_axis_values(None)
                Zret.add_flag('psd_data', True)
            comment = 'PSD 2D along last two axis'

            tot_psd = np.nansum(Zret) / (N[-1] * T[-1] * N[-2] * T[-2])
            tot_z2 = np.nansum(Z ** 2) / (N[-1] * N[-2])
            txt = '{}{} : Sum_psd/Nd = {:.4f}, Sum_z2/N={:.4f}\n'.format(self.label.text(), item, tot_psd, tot_z2)
            self.label.setText(txt)
        else:
            if not any(axes):
                axes = [-1]
            N = Z.shape[axes[-1]]
            T = self.pix_size[-1]
            unit_f = ''
            if isinstance(Z, MetrologyData):
                arr = Z.get_axis_val_items_detector()[-1]
                if np.any(arr.value):
                    T = np.nanmean(np.diff(arr))
                    unit_f = '1/{}'.format(arr.unit)

            w = self.get_window(Z.ndim, N, axes[-1])
            zf = fft(Z*w, axis=axes[-1])
            if isinstance(Z, MetrologyData):
                zf *= Z.unit
            slc = [slice(None)]*Z.ndim
            slc[axes[-1]] = slice(0,N//2)
            Zret = (2.0*T/N)*np.abs(zf[slc])**2
            slc[axes[-1]] = 0
            Zret[slc] = Zret[slc]/2
            xf = fftfreq(N, T)[:N//2]

            if isinstance(Z, MetrologyData):
                pix_size = Z.pix_size_detector
                if not any(pix_size):
                    Z._set_dim_detector([True])
                pix_size = Z.pix_size_detector
                pix_size[-1] = Quantity(np.nanmean(np.diff(xf)), unit=unit_f)
                Zret = Z.copy_to(Zret)
                Zret._set_pix_size(pix_size)
                Zret._set_axis_values(None)
                Zret.add_flag('psd_data', True)
            comment = 'PSD 1D along last axis'

            tot_psd = np.nansum(Zret) / (N * T)
            tot_z2 = np.nansum(Z ** 2) / (N)
            txt = '{}{} : Sum_psd/Nd = {:.4f}, Sum_z2/N={:.4f}\n'.format(self.label.text(), item, tot_psd, tot_z2)
            self.label.setText(txt)
            if self.option==self.AVG_1D and Zret.ndim>1:
                Zret = np.nanmean(Zret, axis=tuple(axes[:-1]))
                comment = 'PSD 1D along last axis and average PSD curves'

        comment += '; Window : {}'.format(self.WINDOWS[self.win])
        return Zret, comment



[docs]    def get_window(self, ndim, N, axis):
        from scipy.signal.windows import blackman, hann, hamming, blackmanharris, kaiser
        w = 1
        if self.win != self.NONE:
            shp = np.ones((ndim,), dtype=int)
            shp[axis] = N
            if self.win == self.BLACKMAN:
                w = blackman(N).reshape(shp)
            elif self.win == self.HANNING:
                w = hann(N).reshape(shp)
            elif self.win == self.HAMMING:
                w = hamming(N).reshape(shp)
            elif self.win == self.BLACKMAN_HARRIS:
                w = blackmanharris(N).reshape(shp)
            elif self.win == self.KAISER:
                w = kaiser(N, beta=self.kaiser_beta).reshape(shp)
        return w
