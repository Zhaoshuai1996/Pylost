[docs]def generate_instr_errors(n_samples, size, pv_herr):
    result = np.zeros((n_samples, 1)+size, dtype=np.float32)
    for i in range(n_samples):
        x = np.arange(size[-1]).reshape(1, -1) - np.random.randint(0, size[-1])
        sig = 0
        for j in range(5):
            f = 2 * (10/(0.1+np.random.rand())) * np.pi/size[-1]
            sig += np.random.rand()*np.sin(f*x)
        sig = sig / pv(sig)
        pv_val = np.random.uniform(pv_herr[0], pv_herr[1]) if isinstance(pv_herr, (list, tuple)) and len(pv_herr)>=2 else pv_herr
        result[i, 0, :, :] += pv_val * sig
    return result



[docs]def get_subap_shape(n_samples, size, option, radius=[0,0]):
    if option == 'flat':
        return 0
    elif option == 'sphere':
        ret = np.zeros((n_samples,1)+size)
        for i in range(n_samples):
            x = np.arange(size[-1]).reshape(1, 1, 1, -1) - np.random.randint(0, size[-1])
            y = np.arange(size[-2]).reshape(1, 1, -1, 1) - np.random.randint(0, size[-2])
            R = 1
            if np.any(radius):
                R = np.power(10, np.random.uniform(radius[0], radius[1]))
            r = np.sqrt(x**2 + y**2) * 1e-3
            h = R - np.sqrt(R**2 - r**2)
            h = h - np.min(h)
            ret[i] = h * 1e9
        return ret



[docs]def get_subap_shape_errors(n_samples, size, nb_subaps, pv_shp_err, min_overlap):
    ret = np.zeros((n_samples,nb_subaps)+size)
    mlen = int(size[1] + (1-min_overlap)*size[1]*nb_subaps)
    sigs = np.zeros((n_samples,1,mlen))
    for t in range(n_samples):
        x = np.arange(mlen).reshape(1, -1) - np.random.randint(0, mlen)
        sig = 0
        for j in range(10):
            f = 2 * (1/(0.01+np.random.rand())) * np.pi/mlen
            sig += np.random.rand()*np.sin(f*x)
        sig = pv_shp_err * sig / pv(sig)
        k = int((1-min_overlap)*size[1])
        of = 0
        for i in range(nb_subaps):
            ret[t,i,:,:] = sig[:, of:of+size[-1]]
            o = np.random.randint(int(k/2), k)
            of += o
        sigs[t] = sig
    return ret, sigs



[docs]def get_retrace_errors(n_samples, heights, scale):
    h = heights - np.min(heights, axis=(-2,-1))[..., np.newaxis, np.newaxis]
    ret = np.zeros_like(heights)
    h = h*scale
    for i in range(n_samples):
        p = np.random.rand()
        ret[i] = (p*np.exp(abs(h[i])) + (1-p)*h[i]**2)
        ret[i] = ret[i] - np.min(ret[i])
    return ret



[docs]def get_mirror_measurements(n_samples=100, nb_subaps=3, size=(64, 64), pv_herr=0.1, pv_shp_err=0.1, min_overlap=0.6, option='flat', base_ref=None,
                            retrace_scale=1e-6, random_std=1, radius_range=[1,4], xscale=-1):
    X = np.zeros((n_samples, nb_subaps)+size, dtype=np.float32)
    instr_err = generate_instr_errors(n_samples, size, pv_herr)
    X += instr_err
    shape = get_subap_shape(n_samples, size=size, option=option, radius=radius_range)
    X += shape
    shp_err, mir_shp_err = get_subap_shape_errors(n_samples, size, nb_subaps, pv_shp_err, min_overlap)
    X += shp_err
    if base_ref is not None:
        X += base_ref[np.newaxis, np.newaxis, :, :]

    retrace_err = 0
    if retrace_scale>0:
        retrace_err = get_retrace_errors(n_samples, shape, retrace_scale)
        X += retrace_err
    X += np.random.normal(scale=random_std, size=X.shape)

    if xscale==-1:
        xscale = 1e3 #(np.max(X) - np.min(X))
    X = X / xscale
    return X, instr_err, xscale, shape, shp_err, mir_shp_err, retrace_err
