[docs]    def stitch1DImage(self):
        """
        Stitching function

        :return:
        """
        start = time.time()                   
        h5stitch                                            = self.h5f[self.stitch_loc]  
        self.updateParams(h5stitch.attrs)
        h5stitch.attrs['date']                              = str(datetime.datetime.now())
        h5stitch.attrs['creator']                           = u'matrix_overlaperr_stitch_fast.py'
        
        self.h5msr                                          = self.h5f[self.msrEntry]
        instrScale                                          = self.h5msr['Instrument/scale_factor'][...] if 'Instrument/scale_factor' in self.h5msr else 1
        h5scans                                             = self.h5msr['Data']                                         
        stitch_step                                         = np.double(h5stitch['stitch_step'][...])  # used when motorX/Y not available
        self.pix_size                                       = np.double(self.h5msr['Instrument/resolution'][...])         #pixel resolution in mm or um
        self.saveMidResults                                 = True if ('save_intermediate_results' in h5stitch.keys() and h5stitch['save_intermediate_results'][...]=='Y') else False

        if self.avg_scanwise_subap:# and ('Scan_avg' not in h5scans.keys()):
            rawScanAvg(h5scans)

        j = 0
        tot_scan_cnt = 0
        for i in h5scans.keys():
            if self.avg_scanwise_subap:
                if i!='Scan_avg': continue
            else:
                if i=='Scan_avg': continue
            if 'NX_class' in h5scans[i].attrs and h5scans[i].attrs['NX_class']=='NXdata': #loop over scans
                try:
                    tot_scan_cnt+=1
                    # self.showPlots=True if i=='Scan_b15' else False
                    if any(self.quser_scans):
                        if i not in self.quser_scans: continue
                    print('Stitching '+i)

                    h5si                        = h5scans[i]
                    h5fi                        = h5stitch.create_group('stitched_scans/'+h5si.attrs['name']) # create stitched folder with same name as data e.g.Scan_b1

                    #######################
                    ## Reading pitch roll errors from calibration
                    if j==0:
                        xpos = h5si['motor_X'][...] # assume same value in repeated scans
                        if self.use_calib_cor_x:
                            fCorX = str(QFileDialog.getOpenFileName(None, 'Open calibration correctors X')[0])
                            cxa = np.loadtxt(fCorX)
                            self.cx_cal = np.interp(xpos, cxa[:, 0], cxa[:, 1]) * 1e-3 *(-1000/instrScale) # TODO: better scaling
                        if self.use_calib_cor_y:
                            fCorY = str(QFileDialog.getOpenFileName(None, 'Open calibration correctors Y')[0])
                            cya = np.loadtxt(fCorY)
                            self.cy_cal = np.interp(xpos, cya[:, 0], cya[:, 1]) * 1e-3 *(-1000/instrScale)
                    #########################

                    #Apply external mask if exists
                    mask                        = h5si['mask'][...]
                    if 'input_mask' in h5stitch:
                        mask                    = applyExternalMask(mask, h5stitch['input_mask'][...])

                    if not np.any(mask):
                        alertMsg('Empty mask','No data selected in the mask')
                        continue

                    # do 1D stitching first
                    mXArr                       = h5si['motor_X'][...]                                  #1D array
                    # mXArr[:]                    = np.nan
                    if np.isnan(np.sum(mXArr)):                                                         #use stitch step if no data is available
                        mXArr                   = range(0,len(mXArr))*stitch_step
                    else:
                        mXArr = -1.0*mXArr if np.all(np.diff(mXArr)<0) else mXArr                       #make sure mottor x is increasing in default forward stitch
                    mXArr_rel                   = mXArr-min(mXArr)                                      #in mm
                    mXArr_rp                    = np.rint(mXArr_rel/self.pix_size[0]).astype(int)       # pixel offsets from start (rp: relative & pixelwise)
                    if self.use_binning:
                        mXArr_rp                = np.rint(mXArr_rel/(self.bin_size_x*self.pix_size[0])).astype(int)
                        mask                    = binData(mask, self.bin_size_x, self.bin_size_y).astype(bool)

                    ########
                    #TODO: interpolation of slp/hgt if step is not integer multiple of pixel
                    szPat                       = mask.shape                                            # patch size XY, last dimension is longest i.e. stitching dir
                    res_szX                     = szPat[-1]+max(mXArr_rp)                               # result size X stitch dirctn
                    res_szY                     = szPat[-2]                                             # result size Y

                    subRefFlag = True if 'ref' in h5si and (not self.h5msr.attrs['flag_ref_subtracted']) else False     # Reference should be subtracted or not?

                    # if self.showPlots:
                    #     plotSt(mask, oxArr=mXArr_rp, pnum=1, title='Mask array')

                    #stitching slopes
                    if self.isSlopeStitch:
                        h5fi, C = self.stitchSlopes(h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale)
                    #stitching height
                    else:
                        h5fi, C = self.stitchHeights(h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale)

                    h5fi['intensity'] = []

                    if self.saveMidResults:
                        h5fi['correctors'] = C.T
                        h5fi['correctors'].attrs[u'format'] = 'num of subapertures x [pitch,roll,piston]'
                        h5fi['correctors'].attrs[u'units'] = h5si['slopes_x'].attrs[u'units'] if self.isSlopeStitch else h5si['height'].attrs[u'units']
                    addAttributesStitchedScans(h5si, h5fi, self.isSlopeStitch, subRefFlag, pix_size = self.pix_size, gravitySubFlag=False)# add attributes for scans
                    j = j+1
                except Exception as e:
                    print('stitch1DImage <- MatrixOverlapErrStitchFast')
                    print(e)
                    # alertMsg('Fail','Failed to stitch')

        self.avgScans(mXArr, tot_scan_cnt)
        print(time.time()-start)
        alertMsg('Finished','Finished stitching')

        # show()
    

[docs]    def stitchSlopes(self, h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale):
        """
        Slopes stitching function on i-th scan

        :param h5si: H5 scan object containing subaperture data of i-th scan
        :param h5fi: H5 stitched object
        :param subRefFlag: Subtract reference flag
        :param mask: Subaperture mask array
        :param mXArr_rp: Translation offset array in pixels
        :param res_szY: Stitched result size Y
        :param res_szX: Stitched result size X
        :param instrScale: Instrument scaling factor
        :return: H5 stitched object (h5fi), correctors
        """
        #####################################
        ## PreProcessing
        # subtract reference
        slxArr = h5si['slopes_x'][...] if not subRefFlag else h5si['slopes_x'][...]-h5si['ref/slopes_x'][...] #ref should be broadcasted to all patches
        slyArr = h5si['slopes_y'][...] if not subRefFlag else h5si['slopes_y'][...]-h5si['ref/slopes_y'][...]
        if self.use_calib_tilt_ref: # Get reference based on tilt calibration
            slxArr, slyArr = subCalRefSlp(self, slxArr, slyArr, -instrScale/1000, h5fi) #TODO: fix instrument scale factor
        if self.use_applied_stage_tilt: # Add tilt values of stages if available
            slxArr, slyArr = addAppliedStageTilt(self, slxArr,slyArr, -instrScale/1000, h5si, h5fi)
        if self.use_binning:
            slxArr                    = binData(slxArr, self.bin_size_x, self.bin_size_y)
            slyArr                    = binData(slyArr, self.bin_size_x, self.bin_size_y)
        # reference from measured data
        if self.useRefMeasureData=='avg':
            slxArr,slyArr = preprocessSlp('REF_M_A',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave=self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
        if self.useRefMeasureData == 'avg_plus_extract':
            slxArr,slyArr = preprocessSlp('REF_M_A_HF',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave=self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
            self.useRefMeasureData = 'extract_from_stitching'
        elif self.useRefMeasureData=='avg_high_freq':
            slxArr,slyArr = preprocessSlp('REF_M_A_HF',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave=self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
        elif self.useRefMeasureData=='high_freq':
            slxArr,slyArr = preprocessSlp('REF_M_HF',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave=self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
            
        # correction for ellipse
        if self.useThreshold=='pre_process':
            slxArr,slyArr = preprocessSlp('E_FS',slxArr,slyArr, args=[self.thrMin, self.thrMax])
        #####################################

        if ~np.any(~np.isnan(slxArr)):
            return h5fi, []
        if ~np.any(~np.isnan(slyArr)):
            return h5fi, []

        Ax, Ex, validOvrX, validPatchesX, slxArr_i = self.buildOverlapErrors(mXArr_rp, slxArr, mask, res_szY, res_szX, otype='sx')
        Ay, Ey, validOvrY, validPatchesY, slyArr_i = self.buildOverlapErrors(mXArr_rp, slyArr, mask, res_szY, res_szX, otype='sy')
        
        minVLen = np.min([len(validOvrX), len(validOvrY)])
        validOvrX[:minVLen] = validOvrX[:minVLen] * validOvrY[:minVLen]
        validOvrY[:minVLen] = validOvrX[:minVLen] * validOvrY[:minVLen]
        validOvrX[minVLen:] = False
        validOvrY[minVLen:] = False
        validPat = validPatchesX*validPatchesY
        if ~np.any(validPat):
            return h5fi, []

        A = Ax[validOvrX]
        Ex = Ex[validOvrX]
        Ey = Ey[validOvrY]
        nPat = A.shape[-1]

        C_ref_x = None
        C_ref_y = None
        if self.oeType=='FULL':
            cx = self.getCorrectorsFullOE(A, Ex, validPat, mXArr_rp)
            cy = self.getCorrectorsFullOE(A, Ey, validPat, mXArr_rp)
            if self.useRefMeasureData == 'extract_from_stitching':
                C_ref_x = cx[nPat:]
                cx = cx[:nPat]
                C_ref_y = cy[nPat:]
                cy = cy[:nPat]
            C = np.stack((cx, cy), axis=1)
        else:
            E = np.stack((Ex, Ey), axis=1) #Ex,Ey are 1D arays stacked to 2D array E
            if A.shape[0]==A.shape[1]: #square
                A_csr = A.tocsr()
                C = spsolve(A_csr, E)
            else:
                invA = np.linalg.pinv(A.toarray())
                C = np.matmul(invA,E)
            
            if self.post_process == 'ref_cv':
                C[validPat,:] = signal.detrend(C[validPat,:], axis=0)

        if self.use_calib_cor_x:
            C[:,0] = self.cx_cal
        if self.use_calib_cor_y:
            C[:,1] = self.cx_cal

        h5fi['slopes_x'] = instrScale*self.correctAndJoin(C[:,0], slxArr_i, otype='sx', h5fi=h5fi, oXArr=mXArr_rp, C_ref=C_ref_x) # scaling -0.5*1e3 is for sharper
        h5fi['slopes_y'] = instrScale*self.correctAndJoin(C[:,1], slyArr_i, otype='sy', h5fi=h5fi, oXArr=mXArr_rp, C_ref=C_ref_y)

        # for sharper ...
        if 'intensity' in h5si and np.any(h5si['intensity'][...]):
            h5fi['intensity_photon_noise'] = self.totalIntensityNoise(h5si['intensity'][...], oXArr=mXArr_rp)
            h5fi['intensity_mean'] = np.nanmean(np.nanmean(h5si['intensity'][...], axis=-1), axis=-1)
        
        return h5fi, C



[docs]    def stitchHeights(self, h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale):
        """
        Height stitching function on i-th scan

        :param h5si: H5 scan object containing subaperture data of i-th scan
        :param h5fi: H5 stitched object
        :param subRefFlag: Subtract reference flag
        :param mask: Subaperture mask array
        :param mXArr_rp: Translation offset array in pixels
        :param res_szY: Stitched result size Y
        :param res_szX: Stitched result size X
        :param instrScale: Instrument scaling factor
        :return: H5 stitched object (h5fi), correctors
        """
        #####################################
        ## PreProcessing
        # subtract reference
        hArr = h5si['height'][...] if not subRefFlag else h5si['height'][...]-h5si['ref/height'][...] #ref should be broadcasted to all patches
        if self.use_binning:
            hArr                    = binData(hArr, self.bin_size_x, self.bin_size_y)
            
        # reference from measured data
        if self.useRefMeasureData=='avg':
            hArr = preprocessHgt('REF_M_A',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
        if self.useRefMeasureData == 'avg_plus_extract':
            hArr = preprocessHgt('REF_M_A_HF', hArr, pix_size=self.pix_size, mask=mask, outH5Obj=h5fi,isSave=self.saveMidResults, showPlots=self.showPlots,nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
            self.useRefMeasureData = 'extract_from_stitching'
        if self.useRefMeasureData=='avg_high_freq':
            hArr = preprocessHgt('REF_M_A_HF',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
        if self.useRefMeasureData=='high_freq':
            hArr = preprocessHgt('REF_M_HF',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults, showPlots=self.showPlots, nbExclEdges=self.use_ref_exclude_nb_edge_subaps)
        #####################################

        print('Fin preprocessing')

        if ~np.any(~np.isnan(hArr)):
            return h5fi, []
        Ar, Er, validOvr, validPat, hArr_i = self.buildOverlapErrors(mXArr_rp, hArr, mask, res_szY, res_szX, otype='h')
        
        if ~np.any(validPat):
            return h5fi, []
        A = Ar[validOvr]
        E = Er[validOvr] # 3 cols pitch/roll/piston
        nPat = A.shape[-1]

        C_ref = None
        if self.oeType=='FULL':
            C = self.getCorrectorsFullOE(A, E, validPat, mXArr_rp)
            if self.useRefMeasureData == 'extract_from_stitching':
                C_ref = C[3*nPat:]
                C = C[:3*nPat]
            C = np.reshape(C, (3,nPat)).T
        else:
            if A.shape[0]==A.shape[1]: #square
                A_csr = A.tocsr()
                C = spsolve(A_csr, E)
            else:
                invA = np.linalg.pinv(A.toarray())
                C = np.matmul(invA,E)
            if self.oeType=='SM':
                C = C[:, np.newaxis]
                C = np.hstack([C, np.zeros_like(C), np.zeros_like(C)])
        print('Fin determining correctors array')

        if self.post_process == 'ref_cv':
            C[validPat,1:3] = signal.detrend(C[validPat,1:3], axis=0)
        if self.showPlots:
            plotSt(C[:,0], pnum=6, title='Piston')
            plotSt(C[:,1], pnum=7, title='Roll')
            plotSt(C[:,2], pnum=8, title='Pitch')

        h5fi['height'] = instrScale*self.correctAndJoin(C, hArr_i, otype='h', h5fi=h5fi, oXArr=mXArr_rp, C_ref=C_ref)

        print('Fin correction')

        return h5fi, C

[docs]    def getCorrectorsFullOE(self,A,E,validPat,oXArr):
        """
        Get correctors when all pixels in the overlap are used

        :param A: IndexMatrix
        :param E: Overlap errors array
        :param validPat: Valid subapertures
        :param oXArr: Translation offset array
        :return: Correctors array
        """
        xvRef, yvRef        = getXYGrid(np.ones_like(E[0], dtype='float'), pix_size=self.pix_size, order=2) # ref is repeated and has shape of (masked) subaperture
        nPat                = A.shape[1]
        nOv                 = E.shape[-3]

        res_mask_szY        = m

        E_full              = E[~np.isnan(E)][:,np.newaxis]
        rowSz               = E_full.size

        if ~self.isSlopeStitch:
            colSz           = 3*nPat
        if self.useRefMeasureData == 'extract_from_stitching':
            # colSz           = colSz + m*n
            B               = None #lil_matrix((rowSz, m*n), dtype=float) # less than m*n*nOv
        A_full              = None #lil_matrix((rowSz, colSz), dtype=float)

        ci = 0
        cf = 0
        for iOv,a in enumerate(A):
            print('Building full matrix : '+str(iOv+1)+'/'+str(A.shape[0]))
            mask = ~np.isnan(E[iOv])
            rsz_i = E[iOv][mask].size

            i = find(a)[1][0] # first nonzero index
            ox = oXArr[i]
            xm = xx_mir[:, ox: ox + E.shape[-1]]
            ym = yy_mir[:, ox: ox + E.shape[-1]]
            a_0 = a.multiply(lil_matrix(np.ones([rsz_i,1], dtype=float))) # broadcasting to size rows(e_i) x cols(a_0) where rows(e_i) = num of pixels in ith overlap, cols(a_0) = num of subap/pathces
            if ~self.isSlopeStitch:
                a_x = a.multiply(xm[mask][:,np.newaxis]) if find(a)[1].size==2 else lil_matrix((rsz_i,a.shape[1]), dtype=float)
                a_y = a.multiply(ym[mask][:,np.newaxis]) if find(a)[1].size==2 else lil_matrix((rsz_i,a.shape[1]), dtype=float)
                a_i = hstack([a_0,a_y,a_x])
            else:
                a_i = a_0
            ci = cf
            cf = ci + rsz_i
            if self.useRefMeasureData == 'extract_from_stitching':
                if find(a)[1].size != 2:
                    exclCurIdx = True
                else:
                    j = find(a==1)[1][0]
                    exclCurIdx = False
                    if self.use_ref_exclude_nb_edge_subaps > 0:
                        exclCurIdx = True if i < self.use_ref_exclude_nb_edge_subaps else False
                        exclCurIdx = exclCurIdx or (True if j >= A.shape[-1] - self.use_ref_exclude_nb_edge_subaps else False)
                if not exclCurIdx:
                    d = oXArr[j] - oXArr[i] #TODO: need to implement for d<0
                    b_d = np.ones((1,m*n))
                    # b_d = mask.astype(int)
                    b_i = diff_k_sp(b_d.ravel()[np.newaxis,:], n=-d, typ='spdiag')
                    b_i = b_i[:-d,:]
                    b_i = b_i[mask.ravel()]
                    # B[ci:cf] = b_i ## it is faster with vstack
                    B = vstack([B,b_i]) if B!=None else b_i
                else:
                    b_i = lil_matrix((rsz_i, m*n), dtype=float)
                    B = vstack([B,b_i]) if B!=None else b_i
            # A_full[ci:cf] = a_i
            A_full = vstack([A_full,a_i]) if A_full!=None else a_i

        if self.useRefMeasureData == 'extract_from_stitching':
            Cxy = self.buildCforExtractRef(xvRef.ravel(), yvRef.ravel()) # use ref coordinate grid
            print('Building AB matrix ... ')
            A_full = hstack([A_full, B*Cxy])

        print('Inverting IndexMatrix ')
        invA = np.linalg.pinv(A_full.toarray())
        print('Calculationg coeficients')
        C = np.matmul(invA, E_full)
        return C

[docs]    def buildCforExtractRefPoly(self,xv,yv):
        """
        Expand index matrix to extract reference from measured data

        :param a: IndexMatrix i-th overlap row
        :param a_i: Pixelwise IndexMatrix i-th overlap rows
        :param xv: X-grid data in reference coordinates
        :param yv: Y-grid data in reference coordinates
        :return: Polynomial term matrix with shape (size of subap valid pixels, num of terms)
        """
        degree = self.degRef
        startDeg = self.startDegRef
        nbTerms = nbTermsPoly(startDeg-1, degree)
        C = np.full((xv.size,nbTerms),np.nan)
        i=0
        for n in range(startDeg, degree + 1):
            for k in range(0, n + 1):
                C[:,i] = xv ** (k) * yv ** (n - k)
                i = i+1
        return C

[docs]    def correctAndJoin(self, c, sarr, otype, h5fi, oXArr, C_ref=None):
        """
        Apply corrections and join the subapertures

        :param c: Correctors
        :param sarr: Subaperture array
        :param otype: Data type - slope/height
        :param h5fi: H5 stitched object
        :param oXArr: Translation offset array
        :return: Stitched image
        """
        res_mask_szY        = sarr.shape[-2]
        res_mask_szX        = sarr.shape[-1]+max(oXArr)
        mask_mir            = np.full([res_mask_szY, res_mask_szX],True)
        xx,yy               = getXYGrid(mask_mir, pix_size=self.pix_size, order=2, mask=mask_mir)
        patch_count,ny,nx   = sarr.shape

        sarr_c = np.full([res_mask_szY, res_mask_szX], np.nan)
        sarr_c_intensity = np.full(sarr_c.shape, 0.0)

        corRef = 0
        if self.useRefMeasureData == 'extract_from_stitching':
            xvRef, yvRef = getXYGrid(np.ones_like(sarr[0], dtype='float'), pix_size=self.pix_size, order=2)
            corRef = self.getCorrectionReference(C_ref, xvRef, yvRef)
            if otype=='h':
                h5fi['reference'] = corRef
            elif otype=='sx':
                h5fi['referenceX'] = corRef
            elif otype == 'sy':
                h5fi['referenceY'] = corRef
        for j,s in enumerate(sarr):
            ox = oXArr[j]
            if otype in ['sx', 'sy']:
                cor = c[j]
            elif otype=='h':
                cor = self.getCorrectionMap(c[j,:],xx[:, ox: ox + nx],yy[:, ox: ox + nx])
            temp = sarr[j, :, :] - cor - corRef
            if self.useThreshold=='post_process':
                temp = self.maskByThreshold(temp, sarr[j, :, :])
            sarr_c[:, ox: ox + nx] = np.nansum(np.dstack((sarr_c[:, ox: ox + nx], temp)),2)
            sarr_c_intensity[:, ox: ox + nx] = sarr_c_intensity[:, ox: ox + nx] + ((~np.isnan(temp)).astype(int))
            if self.showPlots:
                data = np.divide(sarr_c, sarr_c_intensity)
                plotSt(data, pnum=9, title='Stitched profile : '+str(j), pauseTime=0.1)

        sf = np.divide(sarr_c, sarr_c_intensity)
        return sf

[docs]    def getCorrectionReferencePoly(self, c_ref, xx, yy):
        """

        :param c_ref: reference coefficients
        :param xx: reference x grid
        :param yy: reference y grid
        :return: reference map
        """
        corRef = 0
        degree = self.degRef
        startDeg = self.startDegRef
        i = 0
        if self.oeType == 'PF':
            c_ref = c_ref[nbTermsPoly(2, startDeg):]  # ignore coeffs for degree less than startDeg
        for n in range(startDeg, degree + 1):
            for k in range(0, n + 1):
                corRef = corRef + c_ref[i] * xx ** (k) * yy ** (n - k)
                i += 1
        return corRef
