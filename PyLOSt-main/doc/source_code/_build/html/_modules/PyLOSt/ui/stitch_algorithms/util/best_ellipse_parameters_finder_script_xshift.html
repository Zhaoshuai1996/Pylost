[docs]def null_slopes_at_centre(xc, y):
    # X coordinates centered
    size = len(xc)
    mod = size % 2
    if mod > 0: # odd --> no problem
        y = y - y[int(size/2)]
    else: # even --> linear fit over 5 points
        if size > 4:
            first = int(size/2)-2
            last = int(size/2)+2
            y_interp = np.interp((size/2)-0.5, np.arange(first,last), y[first:last])
        else:
            y_interp = (y[int(size/2)] + y[int(size/2)+2]) / 2
        y = y - y_interp

    return y

[docs]def slp_equ_ellipse(x, p, q, theta):
    a = (p+q)/2
    b = np.sqrt(p*q)*np.sin(theta)
#    F = (1/2)*np.sqrt(p**2+q**2-2*p*q*np.cos(np.pi-(2*theta)))
#    alpha = np.arcsin(p/(2*F)*np.sin(np.pi-(2*theta)))
    F = np.sqrt(a**2-b**2)
    alpha = np.arcsin(p/(2*F)*np.sin(2*theta))
    mu = alpha-theta
    x0 = F - q*np.cos(alpha)
    s1 = (np.cos(mu)**2)*(b/a)*((np.cos(mu)*x+x0)/np.sqrt(-1*(np.cos(mu)**2)*(x**2)-2*np.cos(mu)*x*x0+(a**2)-(x0**2)))
    s2 = np.cos(mu)*np.sin(-mu)
    sx = s1 + s2

    return sx

[docs]def sfit_xoffset_loop(x_offset, xc, sx, p_th, q_th, theta_th, optim, method, criteria):

    # new guess of the x_offset
    new_x = xc - x_offset[0]

    # p, q, theta are optimized within boundaries --> TODO: add boundaries
    if method=='min_dq_dtheta':
        p, q, theta, tilt = sfit_curve(new_x, sx, p_th, q_th, theta_th, optim, notilt='false')
        val = ((p_th-p)*1e-3, (q_th-q)*1e-0, ((theta_th*1e-3)-theta)*6.3662e1)

    # finding null tilt from shifted best ellipse
    elif method=='min_tilt':
        # get rid of tilt
        p, q, theta, tilt = sfit_curve(new_x, sx, p_th, q_th, theta_th, optim, notilt='false')
        val = (tilt*1e12)

    else:
        # p, q, theta are not optimized
        if method=='fixed_p_q_theta':
            p, q, theta = p_th, q_th, theta_th
            tilt = 0.0
        # find mininum errors from shifted best ellipse with tilt correction
        if method=='min_shift':
            # errors from last best ellipse
            p, q, theta, tilt = sfit_curve(new_x, sx, p_th, q_th, theta_th, optim, notilt='false')
        # find mininum errors from shifted best ellipse without tilt correction
        if method=='min_best':
            # errors from last best ellipse
            p, q, theta, tilt = sfit_curve(new_x, sx, p_th, q_th, theta_th, optim, notilt='true')

        # errors from theoretical shifted ellipse
        slp_err = sx - slp_equ_ellipse(new_x, p, q, theta)
        if criteria=='PV':
            val = ((p_th-p)*1e0, (q_th-q)*1e0, ((theta_th*1e-3)-theta)*1e3, np.max(slp_err)-np.min(slp_err)*1e0, tilt*0e0)
        elif criteria=='RMS':
            val = ((p_th-p)*1e0, (q_th-q)*1e0, ((theta_th*1e-3)-theta)*1e3, np.std(slp_err)*1e0, tilt*0e0)

#    print(val)
    return val # value to minimize

[docs]def hgt_equ_ellipse(x, p, q, theta):
    a = (p+q)/2
    b = np.sqrt(p*q)*np.sin(theta)
#    F = (1/2)*np.sqrt(p**2+q**2-2*p*q*np.cos(np.pi-(2*theta)))
#    alpha =np.arcsin(p/(2*F)*np.sin(np.pi-(2*theta)))
    F = np.sqrt(a**2-b**2)
    alpha = np.arcsin(p/(2*F)*np.sin(2*theta))
    mu = alpha-theta
    x0 = F - q*np.cos(alpha)
    y0 = -b*np.sqrt(1-(x0/a)**2)
    z1 = np.cos(mu)*(-np.sqrt((b**2)*(1-((x*np.cos(mu)+x0)/a)**2))-y0)
    z2 = np.sin(-mu)*x*np.cos(mu)
    z3 = np.cos(mu)*(y0+np.sqrt((b**2)*(1-(x0/a)**2)))
    hx = z1+z2+z3
    return x0, y0, hx
