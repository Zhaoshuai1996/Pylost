[docs]def diff_k_sp(a, n=2, typ='spdiag', oxArr=None, a_shft=None) : #axis 0
    """
    k-th order difference in an ndarray i.e.difference between i,i+k elements

    :param a: slope/height data array
    :param n: difference order
    :param typ: type of data
    :param oxArr: Translation offset array
    :param a_shft:
    :return: nth order difference array
    """
    if typ=='spdiag':
        diagonals = np.append(-a,a,axis=0) #a is array
        return spdiags(diagonals,[0,n], a.size-n, a.size, format='lil')
    elif typ=='dense':
        return a[n:] - a[:-n]
    elif typ=='overlaps':
        if np.any(oxArr):
            ox_k = diff_k_sp(oxArr, n=n, typ='dense')
            if np.all(ox_k==ox_k[0]):   # constant translation step
                a_shft = np.pad(a,((0,0),(0,0),(ox_k[0],0)),'constant', constant_values=(np.nan,))[:,:,:-ox_k[0]] #pad left side by offset step along X-axis i.e. axis=-1 and slice on the right side by same amount
            else:                       # variable translation step
                if a_shft is None:      a_shft = np.full(a.shape, np.nan, dtype=float)
                else:                   a_shft.fill(np.nan)
                for j,ox in enumerate(ox_k):
                    a_shft[n+j,:,:] = np.pad(a[n+j],((0,0),(ox,0)),'constant', constant_values=(np.nan,))[:,:-ox]
            return a_shft[n:] - a[:-n]
        else: # X-offset already adjusted in the nd array 'a'
            return a[n:] - a[:-n]
    elif typ=='go_errfun':
        if np.any(oxArr):
            ox_k = diff_k_sp(oxArr, n=n, typ='dense')
            if np.all(ox_k==ox_k[0]):   # constant translation step
                a_shft = np.pad(a,((0,0),(ox_k[0],0)),'constant', constant_values=(np.nan,))[:,:-ox_k[0]] #pad left side by offset step along X-axis i.e. axis=-1 and slice on the right side by same amount
            else:                       # variable translation step
                if a_shft is None:      a_shft = np.full(a.shape, np.nan, dtype=float)
                else:                   a_shft.fill(np.nan)
                for j,ox in enumerate(ox_k):
                    a_shft[n+j,:] = np.pad(a[n+j],((ox,0)),'constant', constant_values=(np.nan,))[:-ox]
            return a_shft[n:] - a[:-n]
        else: # X-offset already adjusted in the nd array 'a'
            return a[n:] - a[:-n]



[docs]def filtBadPixels(self, sarr, pix_sz):
    """
    Filter pixels above n-std from global shape.

    :param self: Stitching function reference object
    :param sarr: Input data
    :param pix_sz: Pixel size
    :return: Filter mask
    """
    maxStd                      = float(self.filtBadPixMaxStd)
    sz                          = sarr.shape
    maskBadPix                  = np.full(sarr.shape, False)
    for j in np.arange(0,sz[0]):
        f1,f1resd,_               = fit2D(sarr[j], pix_size=pix_sz, degree=2, retResd=True)
        rmsResd                 = rms(f1resd)
        maskBadPix[j,:,:]       = f1resd < maxStd*rmsResd
    return maskBadPix



[docs]def buildOverlapMatricesFast(self, oXArr, pat_arr, mask_arr, res_szY, res_szX, otype, fitData=True, extractRef=False, degRef=10, refType='legendre', startDegRef=3, showPlots=False):
    """
    Build overlap errors

    :param self: Stitching function reference object
    :param oXArr: Translation offset array
    :param pat_arr: Subaperture array
    :param mask_arr: Mask array of subapertures
    :param res_szY: Stitching result size Y
    :param res_szX: Stitching result size X
    :param otype: Data type slope/height
    :param fitData: Flag to fit plane to overlap
    :param extractRef: Flag to retrieve reference from fitting polynomial
    :param degRef: Polynomial degree to use for extracting reference from fitting
    :param refType: Reference fit type e.g. fit to legendre
    :param startDegRef: Start degree for reference. Usually degree<2 are excluded as theses reference orders cannot be extracted from measured data
    :return: IndexMatrix, overlaps matrix, overalap errors pitch/roll/piston, max overlap count, valid overlaps, valid subapertures, subaperture array, maximum overlap order, piston array
    """
    patch_count = oXArr.size
    max_mask = np.any(mask_arr, axis=0)  # maximum applicable mask elements
    v = np.transpose(np.nonzero(max_mask))
    mn = np.min(v, axis=0)
    mx = np.max(v, axis=0)
    szM = mx - mn + [1, 1]
    validPatches = np.full([patch_count], True)
    pix_sz = self.pix_size

    if otype in ['sx', 'sy', 'h']:
        sarr                = pat_arr[:, mn[-2]:mn[-2] + szM[-2], mn[-1]:mn[-1] + szM[-1]]
        sarr_mask           = mask_arr[:, mn[-2]:mn[-2] + szM[-2], mn[-1]:mn[-1] + szM[-1]]

        if self.nbExclPatches>0:
            sarr[:self.nbExclPatches,:,:] = np.nan
            sarr[-self.nbExclPatches:,:,:] = np.nan
            sarr_mask[:self.nbExclPatches,:,:] = False
            sarr_mask[-self.nbExclPatches:,:,:] = False

        ## Filter: bad pixels above n*std from ideal shape
        if self.filtBadPix:
            maskBadPix      = filtBadPixels(self, sarr, pix_sz)
            sarr_mask       = sarr_mask*maskBadPix
        sarr[~sarr_mask]    = np.nan
        sarr_ret = sarr
        if self.useFullImages:  # keep full image size
            sarr_ret = np.copy(pat_arr)
            if self.applyMaskCorImg:
                sarr_ret = sarr_ret * mask_arr  # TODO: verify if need to multiplied with nan's not 0's

        if showPlots:
            plotSt(sarr, oxArr=oXArr, pnum=4, title='Data after masking')

        res_mask_szY        = sarr.shape[-2]
        res_mask_szX        = sarr.shape[-1]+max(oXArr)
        mask_mir            = np.full([res_mask_szY, res_mask_szX],True)
        xx_mir, yy_mir      = getXYGrid(mask_mir, pix_size=pix_sz, order=2, mask=mask_mir)
        print('Fin prepare data objects')

        if self.removeOutlierPatches:
            # mask of validpatches: exclude outlier patches
            sarr_rms        = np.sqrt(np.nanmean(np.nanmean((sarr) ** 2, axis=2), axis=1))
            med_savg        = np.median(sarr_rms[validPatches])
            Q               = np.abs(sarr_rms - med_savg) < 4 * np.std(sarr_rms[validPatches] - med_savg)  # TODO: use proximity to neighbours instead
            sarr[~Q, :, :]  = np.nan
            sarr_mask[~Q, :, :] = False
            validPatches    = validPatches * Q
            print('Fin remove outlier subapertures')

        # filter before stitching for coeffs 0 order to 2nd order
        sz_sarr             = sarr.shape
        s_mean_piston       = np.nanmean(np.nanmean(sarr, axis=2), axis=1)
        if self.preFiltArr[0]:
            sarr            = sarr - s_mean_piston.reshape(sz_sarr[0], 1, 1)
            print('Fin filter piston')

        # roughly determine max k, for given minoverlap
        max_pix_offset      = int(szM[-1] * (1 - self.minOverlap) / (1 + self.minOverlap))  # max step to remain within min overlap
        max_k               = 1
        while np.any(diff_k_sp(oXArr,max_k,typ='dense') < max_pix_offset):  # max offset satisfied for all steps. need this if multiple steps are used
            max_k += 1
        if np.all(diff_k_sp(oXArr,max_k,typ='dense') > max_pix_offset): max_k = max_k-1

        max_ov_cnt          = int(max_k * patch_count - (max_k * (max_k + 1) / 2))          # sum of series pc-1 + pc-2 + ...+ pc-max_k   #int(patch_count*(patch_count-1)/2) #nc2 combinations of overlaps
        A                   = lil_matrix((max_ov_cnt + 1, patch_count), dtype=float) #np.full([max_ov_cnt + 1, patch_count], 0.0)                   # 1 more for global tilts & piston, A[-1]=1
        sarr_diff           = np.full([max_ov_cnt + 1, sz_sarr[-2], sz_sarr[-1]], np.nan)
        Ep_diff             = np.full([max_ov_cnt + 1, 3 if not extractRef else 3 + nbTermsPoly(2,degRef)], np.nan)     # errors from plane fitting of overlaps
        overlapCoeffArr     = np.full([max_ov_cnt + 1, 1], 0.0)

        k                   = 1               # diff step for overlap i,i+k
        overlapCoeff        = 1
        nbPatEl             = np.sum(np.sum(~np.isnan(sarr), axis=2), axis=1)
        nbPatEl             = nbPatEl.reshape(1,nbPatEl.size)
        AI_arr              = np.ones((1,patch_count))

        print('Start overlap error matrices')
        ci = 0
        cf = 0
        sarr_shift = np.full(sarr.shape, np.nan, dtype=float)
        while overlapCoeff > self.minOverlap and k <= max_k:
            A_k = diff_k_sp(AI_arr, k, typ='spdiag')
            sarr_diff_k = diff_k_sp(sarr, k, typ='overlaps', oxArr=oXArr, a_shft=sarr_shift)
            nbPatElDiag_diff_k = diff_k_sp(nbPatEl, k, typ='spdiag')

            nbOverlapElements_k = np.sum(np.sum(~np.isnan(sarr_diff_k), axis=2), axis=1)
            nbOverlapElements_k = nbOverlapElements_k.reshape(nbOverlapElements_k.size, 1)
            nbElTot_k = np.sum(np.abs(nbPatElDiag_diff_k), axis=1) - nbOverlapElements_k
            overlapCoeff_k = nbOverlapElements_k / nbElTot_k
            overlapCoeff_k[np.isnan(overlapCoeff_k)] = 0
            overlapCoeff = np.max(overlapCoeff_k)  # atleast 1 diff element has overlap larger than minimum overlap criteria

            ci = cf  # first index starts at 0
            cf = ci + A_k.shape[0]  # last index not included in slicing
            A[ci:cf] = A_k  # slicing in axis 0
            sarr_diff[ci:cf] = sarr_diff_k
            overlapCoeffArr[ci:cf] = overlapCoeff_k

            # lets fit the overlap differences
            for j, sd in enumerate(sarr_diff_k):
                if fitData and overlapCoeff_k[j] > self.minOverlap:
                    ox = oXArr[j]
                    xp = xx_mir[:, ox: ox + szM[-1]]
                    yp = yy_mir[:, ox: ox + szM[-1]]
                    fo,foresd,_ = fit2D(sd, pix_size=pix_sz, degree=1, retResd=extractRef, xv=xp, yv=yp)
                    Ep_diff[ci + j, 0:3] = fo
            # for j in np.arange(0,patch_count-k):
            #     i1 = j; i2=j+k
            #     if fitData and overlapCoeff_k[j]>self.minOverlap:
            #         m_i1 = ~np.isnan(sarr_diff_k[i1])
            #         m_i2 = np.roll(m_i1, -oXArr[i2]+oXArr[i1], axis=1)
            #         xp_i1,yp_i1 = getOvParams(sarr_diff_k, i1, oXArr, xx_mir, yy_mir, szM)
            #         xp_i2,yp_i2 = getOvParams(sarr_diff_k, i2, oXArr, xx_mir, yy_mir, szM)
            #         f1,f1resd,_ = fit2D(sarr[i1,:,:], mask=m_i1, pix_size=pix_sz, degree=1,  xv=xp_i1, yv=yp_i1)
            #         f2,f2resd,_ = fit2D(sarr[i2,:,:], mask=m_i2, pix_size=pix_sz, degree=1,  xv=xp_i2, yv=yp_i2)
            #         Ep_diff[ci+j,0:3] = f2-f1
            k += 1

        # global pitch/roll/piston
        A[-1] = 1
        sarr_diff[-1, :szM[-2], :szM[-1]] = 0
        Ep_diff[-1] = 0

        if showPlots:
            plotSt(sarr_diff, oxArr=oXArr, pnum=5, title='Overlaps', pauseTime=0.1)

        validOverlaps = overlapCoeffArr > self.minOverlap  # logical array of valid overlaps (for axis 0)
        hangingPats = np.sum(A[:, validPatches], axis=1)  # remove overlap pair corresponding to invalid patches
        validOverlaps = validOverlaps & (hangingPats == 0)
        validOverlaps[-1] = True  # Global tilts/ piston
        validOverlaps = np.squeeze(np.asarray(validOverlaps)) # shape (z,1) to (z,)
        print('Fin build overlap error matrices')

        return A, sarr_diff, Ep_diff, max_ov_cnt, validOverlaps, validPatches, sarr_ret, max_k, s_mean_piston
