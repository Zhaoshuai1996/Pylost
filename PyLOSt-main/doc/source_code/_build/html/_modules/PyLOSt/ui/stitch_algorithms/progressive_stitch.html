[docs]    def stitch1DImage(self):
        """
        Stitching function

        :return:
        """
        start = time.time()
        h5stitch                            = self.h5f[self.stitch_loc]                                 #stitching loc object
        h5stitch.attrs['stitched_on']       = str(datetime.datetime.now())
        h5stitch.attrs['creator']           = u'progressive_stitch.py'
        self.updateParams(h5stitch.attrs)
        
        self.h5msr                          = self.h5f[self.msrEntry]
        h5scans                             = self.h5msr['Data']                                          #class NXentry object
        stitch_step                         = np.double(self.h5f[self.stitch_loc]['stitch_step'][...])  # used when motorX/Y not available
        self.pix_size                            = np.double(self.h5msr['Instrument/resolution'][...])         #pixel resolution in mm or um
        self.instrScale                          = self.h5msr['Instrument/scale_factor'][...] if 'Instrument/scale_factor' in self.h5msr else 1
        
        mXArr = None
        
        for i in h5scans.keys():
            if 'NX_class' in h5scans[i].attrs and h5scans[i].attrs['NX_class']=='NXdata': #loop over scans
                print('Stitching '+i)
                h5si                        = h5scans[i]
                h5fi                        = h5stitch.create_group('stitched_scans/'+h5si.attrs['name']) # create stitched folder with same name as data e.g.Scan_b1
                
                self.isSlopeStitch = True if 'slopes_x' in h5si else False
                #Apply external mask if exists
                mask_full                        = h5si['mask'][...]
                if 'input_mask' in h5stitch:
                    mask_full                    = applyExternalMask(mask_full, h5stitch['input_mask'][...])
                    
                # do 1D stitching first
                mXArr                       = h5si['motor_X'][...]                                  #1D array
                if np.isnan(np.sum(mXArr)):                                                         #use stitch step if no data is available
                    mXArr                   = range(0,len(mXArr))*stitch_step
                else:
                    mXArr = -1.0*mXArr if np.all(np.diff(mXArr)<0) else mXArr
                mXArr_rel                   = mXArr-min(mXArr)                                      #usually in mm
                mXArr_rp                    = np.rint(mXArr_rel/self.pix_size[0]).astype(int)            # pixel offsets from start# result size Y

                self.mn, self.mx, self.szM = self.getMaxMaskSz(mask_full)
                self.res_szX = self.szM[-1] + max(mXArr_rp)  # result size X stitch dirctn
                self.res_szY = self.szM[-2]
                self.mask = mask_full[:, self.mn[-2]:self.mx[-2] + 1, self.mn[-1]:self.mx[-1] + 1]
                self.mask_full = mask_full
                self.pat_int = self.mask.astype(int)

                subRefFlag = True if 'ref' in h5si and (not self.h5msr.attrs['flag_ref_subtracted']) else False

                self.res_intensity = np.full([self.res_szY, self.res_szX], np.nan)
                self.res_intensity[self.res_intensity == 0] = np.nan

                if self.isSlopeStitch:
                    self.stitchSlopes(h5si,h5fi,subRefFlag,mXArr_rp)
                else:
                    self.stitchHeights(h5si,h5fi,subRefFlag,mXArr_rp)
                h5fi['intensity'] = self.res_intensity
                addAttributesStitchedScans(h5si, h5fi, self.isSlopeStitch, subRefFlag, pix_size=self.pix_size, gravitySubFlag=False)# add attributes for scans
                
        self.avgScans(mXArr)
        print(time.time()-start)
        alertMsg('Finished','Finished stitching')



[docs]    def stitchSlopes(self,h5si,h5fi,subRefFlag,mXArr_rp):
        slxArr                  = h5si['slopes_x'][...] if not subRefFlag else h5si['slopes_x'][...] - h5si['ref/slopes_x'][...]
        slyArr                  = h5si['slopes_y'][...] if not subRefFlag else h5si['slopes_y'][...] - h5si['ref/slopes_y'][...]
        self.res_slx            = np.full([self.res_szY, self.res_szX], np.nan)
        self.res_sly            = np.full([self.res_szY, self.res_szX], np.nan)
        slxArr, slyArr, self.useRefMeasureData = correctRefFromMeasuredData_slp(self, slxArr, slyArr, self.useRefMeasureData, self.mask_full, h5fi)

        totErrVal                                   = 0.0
        for j, ox in enumerate(mXArr_rp):
            errVal = self.corAndAddSubapJ_slp(slxArr, slyArr, j, self.mn, self.mx, ox, self.szM, self.mask[j])
            self.res_intensity[:, ox:ox + self.szM[-1]] = np.nansum([self.res_intensity[:, ox:ox + self.szM[-1]], self.pat_int[j]],axis=0)  # Update intensity
            totErrVal = totErrVal if np.isnan(errVal) else totErrVal + errVal

        # repeat n times
        self.repeatProgStitch_slp(slxArr, slyArr, mXArr_rp, self.mask, iniErr=totErrVal)

        h5fi['slopes_x'] = self.instrScale * np.divide(self.res_slx, self.res_intensity)
        h5fi['slopes_y'] = self.instrScale * np.divide(self.res_sly, self.res_intensity)

        hArr = h5si['height'][...] if not subRefFlag else h5si['height'][...]-h5si['ref/height'][...] #ref should be broadcasted to all patches

        hArr,self.useRefMeasureData = correctRefFromMeasuredData_height(self, hArr, self.useRefMeasureData, self.mask_full, h5fi)

        totErrVal                                   = 0.0
        for j, ox in enumerate(mXArr_rp):
            errVal = self.corAndAddSubapJ_hgt(hArr, j, self.mn, self.mx, ox, self.szM, self.mask[j])
            self.res_intensity[:, ox:ox + self.szM[-1]] = np.nansum([self.res_intensity[:, ox:ox + self.szM[-1]], self.pat_int[j]],axis=0)  # Update intensity
            totErrVal = totErrVal if np.isnan(errVal) else totErrVal + errVal

        # repeat n times
        self.repeatProgStitch_hgt(hArr, mXArr_rp, self.mask, iniErr=totErrVal)

        h5fi['height'] = self.instrScale * np.divide(self.res_h, self.res_intensity)

        # store repeat results to a new array
        if self.isSlopeStitch:
            self.res_slx_rep                = np.full([self.res_szY, self.res_szX], np.nan)
            self.res_sly_rep                = np.full([self.res_szY, self.res_szX], np.nan)
        totErrVal                   = 0.0
        for k in range(0, 5): #Max repeats of 5 times. Do we need it?
            self.res_slx_rep[:,:] = np.nan
            self.res_sly_rep[:,:] = np.nan

            for j, ox in enumerate(mXArr_rp):
                errVal = self.corAndAddSubapJ_slp(slxArr, slyArr, j, self.mn, self.mx, ox, self.szM, mask[j], isRepeat=True)
                totErrVal = totErrVal if np.isnan(errVal) else totErrVal + errVal
            print('totErrVal : ' + str(totErrVal))
            if totErrVal<iniErr: # this run is better, replace the original result with this
                np.copyto(self.res_slx, self.res_slx_rep)
                np.copyto(self.res_sly, self.res_sly_rep)
            else:
                break # break loop immediately??

[docs]    def corAndAddSubapJ_slp(self, slxArr, slyArr, j, mn, mx, ox, szM, mask_j, isRepeat=False):
        pat_slx = slxArr[j]
        pat_sly = slyArr[j]
        pat_slx = pat_slx[mn[-2]:mx[-2] + 1, mn[-1]:mx[-1] + 1]
        pat_sly = pat_sly[mn[-2]:mx[-2] + 1, mn[-1]:mx[-1] + 1]
        res_slopes_x_chunk = np.divide(self.res_slx[:, ox:ox + szM[-1]], self.res_intensity[:, ox:ox + szM[-1]]) if np.any(self.res_intensity) else self.res_slx[:, ox:ox + szM[-1]]
        res_slopes_y_chunk = np.divide(self.res_sly[:, ox:ox + szM[-1]], self.res_intensity[:, ox:ox + szM[-1]]) if np.any(self.res_intensity) else self.res_sly[:, ox:ox + szM[-1]]
        res_slopes_x_chunk[np.isinf(res_slopes_x_chunk)] = np.nan
        res_slopes_y_chunk[np.isinf(res_slopes_y_chunk)] = np.nan

        pat_slx, errSlx = self.corOverlpaErr(pat_slx, res_slopes_x_chunk, mask_j, otype='sx')
        pat_sly, errSly = self.corOverlpaErr(pat_sly, res_slopes_y_chunk, mask_j, otype='sy')
        errVal = errSlx + errSly

        if isRepeat:
            self.res_slx_rep[:, ox:ox + szM[-1]] = np.nansum([self.res_slx_rep[:, ox:ox + szM[-1]], pat_slx], axis=0)
            self.res_sly_rep[:, ox:ox + szM[-1]] = np.nansum([self.res_sly_rep[:, ox:ox + szM[-1]], pat_sly], axis=0)
        else:
            self.res_slx[:, ox:ox + szM[-1]] = np.nansum([self.res_slx[:, ox:ox + szM[-1]], pat_slx], axis=0)
            self.res_sly[:, ox:ox + szM[-1]] = np.nansum([self.res_sly[:, ox:ox + szM[-1]], pat_sly], axis=0)

        return errVal



[docs]    def corAndAddSubapJ_hgt(self, hArr, j, mn, mx, ox, szM, mask_j, isRepeat=False):
        pat_hgt = hArr[j]
        pat_hgt = pat_hgt[mn[-2]:mx[-2] + 1, mn[-1]:mx[-1] + 1]
        res_height_chunk = np.divide(self.res_h[:, ox:ox + szM[-1]], self.res_intensity[:, ox:ox + szM[-1]]) if np.any(self.res_intensity) else self.res_h[:, ox:ox + szM[-1]]
        res_height_chunk[np.isinf(res_height_chunk)] = np.nan

        pat_hgt, errVal = self.corOverlpaErr(pat_hgt, res_height_chunk, mask_j, otype='h', pix_size=self.pix_size)
        if isRepeat:
            self.res_h_rep[:, ox:ox + szM[-1]] = np.nansum([self.res_h_rep[:, ox:ox + szM[-1]], pat_hgt], axis=0)
        else:
            self.res_h[:, ox:ox + szM[-1]] = np.nansum([self.res_h[:, ox:ox + szM[-1]], pat_hgt], axis=0)

        return errVal



[docs]    def getMaxMaskSz(self, mask):
        max_mask = np.any(mask, axis=0)  # maximum applicable mask elements
        v = np.transpose(np.nonzero(max_mask))
        mn = np.min(v, axis=0)
        mx = np.max(v, axis=0)
        szM = mx - mn + [1, 1]
        return mn,mx,szM

[docs]    def corOverlpaErr(self, pat_o, res_o, mask, otype, pix_size=1):
        """
        Correct j-th subaperture to intermediate (j-1)-th stitched result

        :param pat_o: j-th subaperture
        :param res_o: (j-1)-th intermediate stitched result
        :param mask: j-th subaperture mask
        :param otype: Data type slope/height
        :return: Corrected subaperture
        """
        pat_c = pat_o
        if otype in ['sx', 'sy']:
            # find c in M = cI+pat_o-res_o, such that
            if self.useLSOptimize:
                # sum(M^2) is minimized -- least squares optimization
                p0 = 0 # initial guess
                index = mask*~(np.isnan(pat_o) | np.isnan(res_o)) #select not nan values and in the mask
                if (np.any(pat_o[index]) and np.any(res_o[index]) ):
                    p1, C, info, msg, success = optimize.leastsq(self.errfunc_slp, p0, args=(pat_o[index], res_o[index]), full_output=1)
                    pat_c = p1+pat_o
            else:
                cp = -np.nanmean((pat_o-res_o).flatten())
                pat_c = cp+pat_o
        elif otype=='h':
            if self.useLSOptimize:
                p0 = [0,0,0]
                index = mask*~(np.isnan(pat_o) | np.isnan(res_o)) #select not nan values and in the mask
                if (np.any(pat_o[index]) and np.any(res_o[index]) ):
                    mask_subap        = np.ones_like(mask)
                    xx, yy = getXYGrid(pat_o, pix_size=pix_size, order=2, mask=mask_subap)
                    p1, C, info, msg, success = optimize.leastsq(self.errfunc_h, p0, args=(pat_o[index], res_o[index],xx[index],yy[index]), full_output=1)
                    pat_c = p1[0]+p1[1]*xx+p1[2]*yy+pat_o

        return pat_c, rms((pat_c-res_o).flatten())

        return p+t -y
