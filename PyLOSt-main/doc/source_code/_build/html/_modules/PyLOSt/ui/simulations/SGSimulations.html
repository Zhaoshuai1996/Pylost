[docs]    def genSimu(self):
        if self.mLen.text()=='':            alertMsg('Missing data', 'Enter mirror length');return
        else: ml=int(self.mLen.text())
        if self.mWid.text()=='':            alertMsg('Missing data', 'Enter mirror width');return
        else: mw=int(self.mWid.text())
        if self.saLen.text()=='':            alertMsg('Missing data', 'Enter subaperture length');return
        else: sal=int(self.saLen.text())
        if self.saStep.text()=='':            alertMsg('Missing data', 'Enter step');return
        else: sas=int(self.saStep.text())
        
        nbPat = int((ml-sal)/sas)+1
        
        self.ref = []
        self.mHgt = np.zeros((nbPat, mw, sal))
        self.sx = np.full_like(self.mHgt,np.nan)
        self.sy = np.full_like(self.mHgt,np.nan)
        c = np.zeros((nbPat, 3)) # corrrectors
        
        if self.addRef.isChecked():
            if self.refSelect.currentIndex()==0:#Polynomial
                coeffs = re.split(',',self.refCoeffs.text())
                self.mHgt = self.addRefErr(coeffs, self.mHgt)
        
        if self.mirSelect.currentIndex()==0:#Flat
            pass
        elif self.mirSelect.currentIndex()==1:#Sphere
            if self.radCv.text()=='':            alertMsg('Missing data', 'Enter radius of curvature');return
            else: Rc = int(self.radCv.text())
            cv = 1/Rc
            self.mHgt = self.addSphere(cv, self.mHgt)
            
        if self.addPPR.isChecked():
            pPV = 0 if self.pitchPV.text()=='' else int(self.pitchPV.text())
            rPV = 0 if self.rollPV.text()=='' else int(self.rollPV.text())
            pstPV = 0 if self.pistonPV.text()=='' else int(self.pistonPV.text())
            c[:,0] = (1-0.5*np.random.rand(nbPat))*pPV
            c[:,1] = (1-0.5*np.random.rand(nbPat))*rPV
            c[:,2] = (1-0.5*np.random.rand(nbPat))*pstPV
            self.mHgt = self.addStageErr(c, self.mHgt)
        
        self.sx[:,:,:-1] = np.diff(self.mHgt, 1, axis=2)
        self.sy[:,:-1,:] = np.diff(self.mHgt, 1, axis=1)
            
        
        self.h5f['Data/Scan1/height']                               = self.mHgt
        self.h5f['Data/Scan1/height'].attrs['units']                = u'nm'
        self.h5f['Data/Scan1/slopes_x']                             = self.sx
        self.h5f['Data/Scan1/slopes_x'].attrs['units']              = u'urad'
        self.h5f['Data/Scan1/slopes_y']                             = self.sy
        self.h5f['Data/Scan1/slopes_y'].attrs['units']              = u'urad'
        self.h5f['Data/Scan1/correctors']                           = c
        self.h5f['Data/Scan1/correctors'].attrs['units']            = u'[urad,urad,nm]'
        self.h5f['Data/Scan1/correctors'].attrs['format']           = u'[pitch,roll,piston]'
        self.h5f['Data/Scan1/motor_X']                              = np.arange(0,int(ml-sal)+sas,sas)
        self.h5f['Data'].attrs['flag_ref_subtracted']               = False
        self.h5f['Data/Scan1'].attrs['NX_class']                    = u'NXdata'
        self.h5f['Data/Scan1'].attrs['name']                        = u'Scan1'
        self.h5f['Data/Scan1/mask']                                 = np.ones((nbPat, mw, sal),dtype=bool)
        self.h5f['Instrument/resolution']                           = [1,1]
        self.h5f['Instrument'].attrs[u'instr_id']                   = "FIZ1E"



[docs]    def addRefErr(self, coeffs, sarr, otype='h'):
        deg_x = 0
        deg_y = 0
        refErr = 0
        pix_size                                              = np.array([1,1])
        xx,yy = getXYGrid(sarr,  pix_size)
        for cij in coeffs:
            refErr = refErr + float(cij)*(xx**deg_x)*(yy**deg_y)
            if deg_y == deg_x:
                deg_x = deg_x+1
                deg_y = 0
            else:
                deg_y = deg_y+1
        imshow(refErr)
        show()
        self.h5f['Data/ref'] = refErr
        sarr_c = sarr + refErr
        return sarr_c



[docs]    def addStageErr(self, c, sarr, otype='h'):
        pix_size                                               = np.array([1,1])
        if otype=='h':    
            patch_count,ny,nx           = sarr.shape
            xx,yy                       = getXYGrid(sarr,  pix_size)
        
            corPitch    = c[:,0].reshape(patch_count,1,1)*xx
            corRoll     = c[:,1].reshape(patch_count,1,1)*yy
            corPiston   = c[:,2].reshape(patch_count,1,1)
           
            sarr_c = sarr + (corPitch + corRoll + corPiston)
            return sarr_c

        

[docs]    def addSphere(self, cv, sarr, otype='h'):
        pix_size                                               = np.array([1,1])
        if otype=='h':    
            xx,yy = getXYGrid(sarr,  pix_size)
           
            sarr_c = sarr + cv*xx**2 + cv*yy**2
            return sarr_c
