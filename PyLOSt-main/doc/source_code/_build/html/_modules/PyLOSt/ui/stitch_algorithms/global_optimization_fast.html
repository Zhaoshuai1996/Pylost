[docs]    def stitch1DImage(self):
        """
        Stitching function

        :return:
        """
        start = time.time()
        h5stitch                            = self.h5f[self.stitch_loc]    
        self.updateParams(h5stitch.attrs)
        h5stitch.attrs['stitched_on']                   = str(datetime.datetime.now())
        h5stitch.attrs['creator']                       = u'global_optimization.py'
          
        self.h5msr                          = self.h5f[self.msrEntry]
        h5scans                             = self.h5msr['Data']                                          #class NXentry object                           
        stitch_step                         = np.double(self.h5f[self.stitch_loc]['stitch_step'][...])  # used when motorX/Y not available
        self.pix_size                       = np.double(self.h5msr['Instrument/resolution'][...])         #pixel resolution in mm or um
        instrScale                          = self.h5msr['Instrument/scale_factor'][...] if 'Instrument/scale_factor' in self.h5msr else 1
        self.saveMidResults                 = True if ('save_intermediate_results' in h5stitch.keys() and h5stitch['save_intermediate_results'][...]=='Y') else False
        
        
        for i in h5scans.keys():
            if 'NX_class' in h5scans[i].attrs and h5scans[i].attrs['NX_class']=='NXdata': #loop over scans
                print('Stitching '+i)
                h5si                        = h5scans[i]
                h5fi                        = h5stitch.create_group('stitched_scans/'+h5si.attrs['name']) # create stitched folder with same name as data e.g.Scan_b1
                
#                 self.isSlopeStitch = True if 'slopes_x' in h5si else False
                mask                        = h5si['mask'][...]
                if 'input_mask' in h5stitch:
                    mask                    = applyExternalMask(mask, h5stitch['input_mask'][...])
                    
                # do 1D stitching first
                mXArr                       = h5si['motor_X'][...]                                  #1D array
                if np.isnan(np.sum(mXArr)):                                                         #use stitch step if no data is available
                    mXArr                   = range(0,len(mXArr))*stitch_step
                else:
                    mXArr = -1.0*mXArr if np.all(np.diff(mXArr)<0) else mXArr
                mXArr_rel                   = mXArr-min(mXArr)                                      #usually in mm
                mXArr_rp                    = np.rint(mXArr_rel/self.pix_size[0]).astype(int)          # pixel offsets from start (rp: relative & pixelwise)
                ########
                #TODO: interpolation of slp/hgt if step is not integer multiple of pixel
                szPat                       = mask.shape                                            # patch size XY, last dimension is longest i.e. stitching dir
                res_szX                     = szPat[-1]+max(mXArr_rp)                               # result size X stitch dirctn
                res_szY                     = szPat[-2]                                             # result size Y
                
                subRefFlag = True if 'ref' in h5si and (not self.h5msr.attrs['flag_ref_subtracted']) else False
                
                #########################################
                #stitching slopes
                if self.isSlopeStitch:
                    # subtract reference
                    slxArr = h5si['slopes_x'][...] if not subRefFlag else h5si['slopes_x'][...]-h5si['ref/slopes_x'][...] #ref should be broadcasted to all patches
                    slyArr = h5si['slopes_y'][...] if not subRefFlag else h5si['slopes_y'][...]-h5si['ref/slopes_y'][...]
                    slxArr, slyArr, self.useRefMeasureData = correctRefFromMeasuredData_slp(self, slxArr, slyArr, self.useRefMeasureData, mask, h5fi)
                    
                    Cx, validPatchesX, slxArr_i = self.buildOverlapErrors(mXArr_rp, slxArr, mask, res_szY, res_szX, otype='sx')
                    Cy, validPatchesY, slyArr_i = self.buildOverlapErrors(mXArr_rp, slyArr, mask, res_szY, res_szX, otype='sy')
                 
                    C = np.concatenate((Cx,Cy),axis=1)
                    validPat = validPatchesX*validPatchesY
                    
                    if self.post_process == 'ref_cv':
                        C[validPat,:] = signal.detrend(C[validPat,:], axis=0)
                        
                    h5fi['slopes_x'] = instrScale*self.correctAndJoin(C[:,0], slxArr_i, otype='sx', oXArr=mXArr_rp) # scaling -0.5*1e3 is for sharper
                    h5fi['slopes_y'] = instrScale*self.correctAndJoin(C[:,1], slyArr_i, otype='sy', oXArr=mXArr_rp)
                #########################################
                #stitching height
                else:
                    # subtract reference
                    hArr = h5si['height'][...] if not subRefFlag else h5si['height'][...]-h5si['ref/height'][...] #ref should be broadcasted to all patches
                    hArr,self.useRefMeasureData = correctRefFromMeasuredData_height(self, hArr, self.useRefMeasureData, mask, h5fi)

                    C, validPat, hArr_i = self.buildOverlapErrors(mXArr_rp, hArr, mask, res_szY, res_szX, otype='h')
                    
                    #TODO: correctors calculation for height
                    h5fi['height'] = instrScale*self.correctAndJoin(C, hArr_i, otype='h', oXArr=mXArr_rp)
                    
                h5fi['intensity'] = []

                if self.saveMidResults:
                    h5fi['correctors'] = C.T
                    h5fi['correctors'].attrs[u'format'] = 'num of subapertures x [pitch,roll,piston]'
                    h5fi['correctors'].attrs[u'units'] = h5si['slopes_x'].attrs[u'units'] if self.isSlopeStitch else h5si['height'].attrs[u'units']

                addAttributesStitchedScans(h5si, h5fi, self.isSlopeStitch, subRefFlag, pix_size = self.pix_size, gravitySubFlag=False)# add attributes for scans

        self.avgScans(mXArr)
        print(time.time()-start)
        alertMsg('Finished','Finished stitching')

        # show()

[docs]    def buildOverlapErrors(self, oXArr, pat_arr, mask_arr, res_szY, res_szX, otype):
        """
        Overlap errors are retrieved as numpy ndarray and correctors are calculated

        :param oXArr: Translation offset array in pixels
        :param pat_arr: Subaperture slope/height array
        :param mask_arr: Mask array with same shape as pat_arr
        :param res_szY: Stitched result data size Y
        :param res_szX: Stitched result data size X
        :param otype: Sata object type, 'sx' for slopes X, 'sy' for slopes Y, 'h' for height
        :return: Correctors, valid subapertures, subaperture data array
        """
        patch_count = len(oXArr)
        A, sarr_diff, Ep_diff, max_ov_cnt, validOverlaps, validPatches, sarr_ret, max_k, s_mean_piston = buildOverlapMatricesFast(self, oXArr, pat_arr, mask_arr, res_szY, res_szX, otype, fitData=self.initAdv)
        
        A = A[validOverlaps]
        sarr_diff = sarr_diff[validOverlaps]
        if self.initAdv:
            Ep_diff = Ep_diff[validOverlaps]
            Ep_diff = Ep_diff if otype=='h' else Ep_diff[:,0]
            # if self.oeType == 'SIMPLE' and otype in ['sx', 'sy']:
            #     Ep_diff = np.nanmean(np.nanmean(sarr_diff, axis=2), axis=1)
            if A.shape[0]==A.shape[1]: #square
                A_csr = A.tocsr()
                C = spsolve(A_csr, Ep_diff)
            else:
                invA = np.linalg.pinv(A.toarray())
                C = np.matmul(invA,Ep_diff)
            if otype == 'h':
                C = np.flip(C,axis=-1)      #cx+by+a
            if otype in ['sx', 'sy']:
                C = C[:,np.newaxis]
        else:
            nc = 3 if otype=='h' else 1
            C = np.full([patch_count,nc], 0, dtype='float32')
            if otype=='h': C[:,2] = s_mean_piston
            else: C = s_mean_piston[:,np.newaxis]

        A = A[:,validPatches]
        C[validPatches] = self.getCorrectors(-C[validPatches], A, sarr_diff, sarr_ret, otype, validOverlaps, max_k, max_ov_cnt, oXArr)
                
        return C, validPatches, sarr_ret

    

[docs]    def getCorrectors(self, P0, A, sarr_diff,sarr_ret, otype, validOverlaps, max_k, max_ov_cnt, oXArr):
        """
        Optimize for correctors where overlap error between subaperture is minimum

        :param P0: Initial correctors
        :param A: Matrix containing subapertures indices in a overlap - IndexMatrix
        :param sarr_diff: Overlap error array
        :param sarr_ret: Subaperture array
        :param otype: Data type slope(X/Y)/height
        :param validOverlaps: Valid overlaps
        :param max_k: Maximum order of overlap difference i.e. there exist a overlap between j, j+k subapertures satisfying minimum overlap condition
        :param max_ov_cnt: Maxim number of overlaps for a given max_k. Actual number may be less if the translation step is not constant
        :return: Optimized correctors
        """
        # find c in M = dcI+dp - for slopes
        # find c in M = daX+dbY+dcI+dp - for heights  -- dp = patch_i-patch_j
        # sum(M^2) is minimized -- least squares optimization
        if self.method=='LS':
            if self.use_binning_opt:
                sarr_diff                   = binData(sarr_diff, self.bin_size_x, self.bin_size_y)
            sarr_diff_px                    = sarr_diff.transpose(0,2,1).reshape(sarr_diff.shape[0],-1)
                
            if otype in ['sx', 'sy']:
                if self.oeType=='SIMPLE':
                    sarr_diff_px                = np.nanmean(sarr_diff_px, axis=1)
                    sarr_diff_px                = sarr_diff_px.reshape(len(sarr_diff_px),1)
                P, C, info, msg, success        = optimize.leastsq(self.errfunc_slp, P0, args=(A, sarr_diff_px), full_output=1) #P0,P column vectors - correctors for pitch or roll
                print('number of function calls =', info['nfev'])
                return -P.reshape(len(P),1)
            elif otype=='h':
                xx,yy                       = self.getXY(sarr_ret, self.pix_size, oXArr)
                if self.use_binning_opt:
                    xx                      = binData(xx, self.bin_size_x, self.bin_size_y)
                    yy                      = binData(yy, self.bin_size_x, self.bin_size_y)
                xx_px                       = xx.transpose(0,2,1).reshape(xx.shape[0],-1)
                yy_px                       = yy.transpose(0,2,1).reshape(yy.shape[0],-1)
                #P0[:,2] = - P0[:,2]
                # P, C, info, msg, success    = optimize.leastsq(self.errfunc_h, P0.flatten(), args=(P0.shape, A, sarr_diff_px, xx_px, yy_px, validOverlaps, max_k, max_ov_cnt), full_output=1) #P0,P- patch_cnt x 3 - for pitch & roll & piston correctors
                # print('number of function calls =', info['nfev'])
                mat_shift = np.full(xx_px.shape, np.nan, dtype=float)
                oXArr_flat = oXArr*xx.shape[1] # shift multiplied column size, to calculated shift in index
                if self.use_binning_opt:
                    oXArr_flat = ((oXArr/self.bin_size_x).astype(int))*xx.shape[1]
                PX_d    = np.full([max_ov_cnt+1, xx_px.shape[1]], np.nan)
                PY_d    = np.full([max_ov_cnt+1, xx_px.shape[1]], np.nan)
                P = optimize.least_squares(self.errfunc_h, x0=P0.flatten(), loss='arctan', args=(P0.shape, A, sarr_diff_px, xx_px, yy_px, validOverlaps, max_k, max_ov_cnt, oXArr_flat, mat_shift, PX_d, PY_d))
                res = -P.x.reshape(P0.shape)
                res = np.flip(res,axis=1)
                res[:,0] = res[:,0]-np.nanmean(res[:,0]) # piston
                res[:,1] = res[:,1]-np.nanmean(res[:,1]) # roll
                res[:,2] = res[:,2]-np.nanmean(res[:,2]) # pitch
                return res
        return P0

        

[docs]    def getXY(self, sarr, pix_sz, oXArr):
        """
        Get XY grid of subapertures in mirror coordinates

        :param sarr:
        :param pix_sz:
        :param oXArr:
        :return:
        """
        res_mask_szY    = sarr.shape[-2]
        res_mask_szX    = sarr.shape[-1] + max(oXArr)
        mask_mir        = np.full([res_mask_szY, res_mask_szX], True)
        xx, yy          = getXYGrid(mask_mir, pix_size=pix_sz, order=2, mask=mask_mir)
        pcnt, ny, nx    = sarr.shape
        xx_arr          = np.zeros_like(sarr)
        yy_arr          = np.zeros_like(sarr)
        for j, s in enumerate(sarr):
            ox = oXArr[j]
            xx_arr[j,:,:] = xx[:, ox: ox + nx]
            yy_arr[j,:,:] = yy[:, ox: ox + nx]
        return xx_arr, yy_arr



[docs]    def errfunc_slp(self, P, A, dS):
        """
        Error function to be optimized for slope data

        :param P: Corrector array
        :param A: IndexMatrix
        :param dS: Slope difference
        :return:
        """
        AP = A.dot(P)
        AP = AP[:,np.newaxis]
        err_F = AP + dS
        print('rms(err_F):'+str(rms(err_F)))
        return err_F[~np.isnan(err_F)]

    

[docs]    def errfunc_h(self, P, sh_P, A, dS, xx_p, yy_p, validOverlaps, max_k, max_ov_cnt, oXArr, mat_shift, PX_d, PY_d):
        """
        Error function to be optimized for height data

        :param P: Corrector array flattened
        :param sh_P: Shape of correctors
        :param A: IndexMatrix
        :param dS: Height difference
        :param xx_p: X-grid values for each overlap
        :param yy_p: Y-grid values for each overlap
        :param validOverlaps: Valid overlaps
        :param max_k: Maximum order of overlap difference i.e. there exist a overlap between j, j+k subapertures satisfying minimum overlap condition
        :param max_ov_cnt: Maxim number of overlaps for a given max_k. Actual number may be less if the translation step is not constant
        :param oXArr: Translation offset array
        :param mat_shift:
        :param PX_d:
        :param PY_d:
        :return:
        """
        #AP = np.matmul(A,P.reshape(sh_P))
        P = P.reshape(sh_P)
        PX = P[:,0].reshape(P.shape[0],1) * xx_p
        PY = P[:,1].reshape(P.shape[0],1) * yy_p
        k=1
        ci = 0
        cf = 0
        while k<=max_k:
            PX_k                            = diff_k_sp(PX,k, typ='go_errfun', oxArr=oXArr, a_shft=mat_shift)
            PY_k                            = diff_k_sp(PY,k, typ='go_errfun', oxArr=oXArr, a_shft=mat_shift)
            ci                              = cf                    # first index starts at 0
            cf                              = ci+PX_k.shape[0]       # last index not included in slicing
            PX_d[ci:cf]                     = PX_k
            PY_d[ci:cf]                     = PY_k
            k+=1
        PX_d                                = PX_d[validOverlaps]
        PY_d                                = PY_d[validOverlaps]
        PX_d[-1]                             = 0
        PY_d[-1]                             = 0
        # PX_d = A.dot(PX)#np.matmul(A,PX)
        # PY_d = A.dot(PY)#np.matmul(A,PY)
        P_d = A.dot(P[:,2].reshape(P.shape[0],1))
        err_F = PX_d + PY_d + P_d
        err_F = err_F + dS
        print('rms(err_F):'+str(rms(err_F)))
        return err_F[~np.isnan(err_F)]

[docs]    def correctAndJoin(self, c, sarr, otype, oXArr):
        """
        Apply corrections and join the subapertures

        :param c: Correctors
        :param sarr: Subaperture array
        :param otype: Data type - slope/height
        :param oXArr: Translation offset array
        :return: Stitched image
        """
        res_mask_szY = sarr.shape[-2]
        res_mask_szX = sarr.shape[-1] + max(oXArr)
        mask_mir = np.full([res_mask_szY, res_mask_szX], True)
        xx, yy = getXYGrid(mask_mir, pix_size=self.pix_size, order=2, mask=mask_mir)
        patch_count, ny, nx = sarr.shape

        sarr_c = np.full([res_mask_szY, res_mask_szX], np.nan)
        sarr_c_intensity = np.full(sarr_c.shape, 0.0)

        for j, s in enumerate(sarr):
            ox = oXArr[j]
            if otype in ['sx', 'sy']:
                cor = c[j]
            elif otype == 'h':
                cor = self.getCorrectionMap(c[j, :], xx[:, ox: ox + nx], yy[:, ox: ox + nx])
            temp = sarr[j, :, :] - cor
            if self.useThreshold == 'post_process':
                temp = self.maskByThreshold(temp, sarr[j, :, :])
            sarr_c[:, ox: ox + nx] = np.nansum(np.dstack((sarr_c[:, ox: ox + nx], temp)), 2)
            sarr_c_intensity[:, ox: ox + nx] = sarr_c_intensity[:, ox: ox + nx] + ((~np.isnan(temp)).astype(int))

        sf = np.divide(sarr_c, sarr_c_intensity)
        return sf



[docs]    def getCorrectionMap(self, cj, xx_j, yy_j):
        """
        Get correcion map for j'th subaperture

        :param cj: jth corrector
        :param xx_j: X-grid postions
        :param yy_j: Y-grid postions
        :return:
        """
        if self.oeType == 'SIMPLE':
            corPitch = 0
            corRoll = 0
            corPiston = cj[0]
        else:
            corPitch = cj[2] * xx_j
            corRoll = cj[1] * yy_j
            corPiston = cj[0]

        cor = 0
        cor = cor + corPitch + corRoll + corPiston
        return cor
