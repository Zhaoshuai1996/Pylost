[docs]    def instrClick(self, item):
        cit = item.data(1)#.toInt()[0])
        if cit==0:
            self.hideDetails()
            self.btnAdd.show()
        else:
            self.hideDetails()
            self.btnSave.show()
            self.btnDelete.show()
            self.btnSetupOptions.show()
        self.instrDetails.show()
        # fill details
        if Instruments.selectBy(id=cit).count()>0:
            i = Instruments.selectBy(id=cit)[0]
            self.setCurData(i)
            self.insId.setReadOnly(True)
        else:
            self.clearData()
            self.insId.setReadOnly(False)

[docs]    def generateStitchAlgo(self):
        try:
            if questionMsg(title="Generate algorithm", msg="This creates {function name}.py file in"
                                                           "'/PyLOSt/stitsch_algorithms/' folder. "
                                                           "Do you want to continue (Yes/No?)"):
                self.getCurDataAlgo()
                f_path = os.path.join('stitch_algorithms',self.afunName+'.py')
                base_path = os.path.abspath(os.path.dirname(PyLOSt.__file__))
                f_path = os.path.join(base_path, f_path)
                if os.path.isfile(f_path):
                    alertMsg('File exist','Algorithm already exist')
                else:
                    f = open(f_path,'w+') # create file if it does not exist
                    f.write('class '+self.afunName+':\n')
                    self.qopts = StitchSetupAlgoOptions.selectBy(algoID=self.algoList.currentItem().data(1))
                    for it in self.qopts:
                        f.write('\t'+it.option+'=None\n')
                        #TODO: set algorithm option to their default values
                    f.write('\tdef updateParams(self,h5StAttrs):\n')
                    for it in self.qopts:
                        f.write('\t\tif "'+it.option+'" in h5StAttrs: self.'+it.option+' = h5StAttrs["'+it.option+'"]\n')
                    f.write('\t\tif "measurement_entry" in h5StAttrs: self.measurement_entry = h5StAttrs["measurement_entry"]\n')
                    f.write('\tdef stitch(self):\n')
                    f.write('\t\t#Write your code here')
                    f.close()
        except Exception as e:
            print('generateStitchAlgo <- SGSettings')
            print(e)
