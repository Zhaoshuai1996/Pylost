[docs]    def loadForm(self):
        self.qinsopts = StitchSetupOptionsInstr.selectBy(instrID=Instruments.selectBy(instrId=self.iID)[0].id)
        self.formGroupBox   = QGroupBox("Stitching Options")
        layout              = QFormLayout()
        lbl                 = QLabel('Setup Name')
        self.leSetupName    = QLineEdit(self.msrEntry.strip('/')+'_xx')
        layout.addRow(lbl, self.leSetupName)
        lbl                 = QLabel('Measurement Data')
        self.leMsrEntry    = QLineEdit(self.msrEntry)
        layout.addRow(lbl, self.leMsrEntry)
        self.eArr=[]
        for item in self.qinsopts:
            copt = StitchSetupOptionsCommon.selectBy(id=item.optionID)[0]
            lblTxt = copt.option
            if item.defValUnit!='':
                lblTxt += '('+item.defValUnit+')'
            lblDesc = copt.optionDesc
            
            lbl = QLabel(lblTxt)
            lbl.setToolTip(lblDesc)
            
            leditTxt = item.defVal
            if copt.option=='stitch_algorithm':
                self.algos = QComboBox()
                self.qalgos = Algorithms.select()
                
                self.curAlgoId = -1
                self.algos.addItem('Select algorithm', self.curAlgoId)
                for it in self.qalgos:
                    self.algos.addItem(it.algoName, it.id) #+' [ '+it.algoDesc+' ]'
                self.algos.currentIndexChanged.connect(self.algoChange)
                
                #le = QLineEdit(leditTxt)
                self.eArr.append(self.algos)
                layout.addRow(lbl, self.algos)
            else:
                try:
                    if copt.option == 'stitch_step':
                        mX = np.array([])
                        if 'Data/Scan_1/motor_X' in self.h5Objs[0][self.msrEntry]:
                            mX = self.h5Objs[0][self.msrEntry]['Data/Scan_1/motor_X']
                        if 'Data/Scan_f1/motor_X' in self.h5Objs[0][self.msrEntry]:
                            mX = self.h5Objs[0][self.msrEntry]['Data/Scan_f1/motor_X']
                        if np.any(mX):
                            leditTxt = str(np.abs(np.nanmean(np.diff(mX).flatten())))
                except Exception as e:
                    print("Exception while retrieving step from motor_X")
                    print(e)

                le = QLineEdit(leditTxt)
                self.eArr.append(le)
                layout.addRow(lbl, le)

        self.formGroupBox.setLayout(layout)
        self.formBoxLayout.addWidget(self.formGroupBox)
        
        self.formAlgoOptions = QGroupBox("Algorithm Options")
        self.layoutAlgoOptions = QFormLayout()
        self.formAlgoOptions.setLayout(self.layoutAlgoOptions)
        scroll = QScrollArea()
        scroll.setWidget(self.formAlgoOptions)
        scroll.setWidgetResizable(True)
        self.formBoxLayout.addWidget(scroll)

    

[docs]    def addSetup(self):
        # valiation
        if self.curAlgoId==-1: alertMsg('Add Stitch Setup', 'Please select an algorithm')
        for i,it in enumerate(self.qalgoopts):
            if it.option=='data_type' and self.aArr[i].currentText()=='':
                alertMsg('Algorithm options', 'Please select data_type')
                return

        stitchPath = 'StitchResults/'+self.leSetupName.text()
        while stitchPath in self.h5Objs[0]:
            alertMsg('Setup name already exist', 'Please change setup name')
            return
        
        setup                                                                   = self.h5Objs[0].create_group(stitchPath)
        self.h5Objs[0]['StitchResults'].attrs[u'NX_class']                      = u'NXentry'
        setup.attrs[u'NX_class']                                                = u'NXprocess'
        setup.attrs[u'measurement_entry']                                       = self.leMsrEntry.text()
        setup[u'measurement_entry']                                             = self.leMsrEntry.text()
        
#         qinsopts        = StitchSetupOptionsInstr.selectBy(instrID=Instruments.selectBy(instrId=self.iID)[0].id)
        for i,it in enumerate(self.qinsopts):
            copt                                       = StitchSetupOptionsCommon.selectBy(id=it.optionID)[0]
            if copt.option=='stitch_algorithm':
                calgo                                  = Algorithms.selectBy(id=self.curAlgoId)[0]
                setup[copt.option]                     = calgo.algoName
            else:
                setup[copt.option]                     = self.eArr[i].text() #copt.defVal
            setup[copt.option].attrs[u'units']         = copt.defValUnit
        
        # Algorithms options
        for i,it in enumerate(self.qalgoopts):
            if self.aArrType[i] == 'E':         setup.attrs[it.option]                 = self.aArr[i].text()
            elif self.aArrType[i] == 'C':       setup.attrs[it.option]                 = self.aArr[i].isChecked()
            elif self.aArrType[i] == 'S':       setup.attrs[it.option]                 = self.aArr[i].currentText()
            else :                              setup.attrs[it.option]                 = self.aArr[i].text()

        self.parent.syncH5Obj(setup)
        self.parent.leSaveMask.setText('/'+stitchPath)
        self.close()

[docs]    def updateAlgoLayout(self):
        self.clearLayout(self.layoutAlgoOptions)
        #self.qalgoopts = StitchSetupAlgoOptions.selectBy(algoID=self.curAlgoId)
        self.qalgoopts = StitchSetupAlgoOptions.select(StitchSetupAlgoOptions.q.algoID == self.curAlgoId, orderBy=StitchSetupAlgoOptions.q.id)
        
        self.aArr=[]
        self.aArrType=[]
        for item in self.qalgoopts:
            lblTxt = item.option
            lblDesc = item.optionDesc
            leditTxt = item.defVal
            
            lbl = QLabel(lblTxt)
            lbl.setToolTip(lblDesc)
            
            disp_item = InputDispTypes.selectBy(id=item.dispTypeID)[0]
            if disp_item.dispType=='E': #Lineedit
                le = QLineEdit(leditTxt)
                self.layoutAlgoOptions.addRow(lbl, le)
            elif disp_item.dispType=='C': #Checkbox
                le = QCheckBox('')
                le.setChecked(True if item.defVal=='Y' else False)
                self.layoutAlgoOptions.addRow(lbl, le)
            elif disp_item.dispType=='S': #Selectbox
                le = QComboBox()
                le.addItems(re.split(',',item.allVals))
                self.layoutAlgoOptions.addRow(lbl, le)
            else:
                le = QLineEdit(leditTxt)
                self.layoutAlgoOptions.addRow(lbl, le)
        
            self.aArr.append(le)
            self.aArrType.append(disp_item.dispType)

        d = QDialog()
        formGroupBox    = QGroupBox("Stitching Options", d)
        layout          = QFormLayout()
        for item in self.qinsopts:
            copt = StitchSetupOptionsCommon.selectBy(id=item.optionID)[0]
            lblTxt = copt.option
            if item.defValUnit!='':
                lblTxt += ' ('+item.defValUnit+')'
            lblTxt += ' : '
            lblDesc = copt.optionDesc
            lbl1 = QLabel(lblTxt)
            lbl2 = QLabel(lblDesc)
            layout.addRow(lbl1, lbl2)
            if copt.option=='stitch_algorithm':
                qalgos = Algorithms.select()
                for it in qalgos:
                    lbl3 = QLabel(' - '+it.algoName)
                    lbl4 = QLabel(it.algoDesc)
                    layout.addRow(lbl3, lbl4)

        formGroupBox.setLayout(layout)
        d.setWindowTitle("Stitching options dialog")
        d.resize(800,400)
        d.exec_()
