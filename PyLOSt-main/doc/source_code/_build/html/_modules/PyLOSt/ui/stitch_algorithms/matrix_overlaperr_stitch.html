[docs]    def stitch1DImage(self):
        """
        Stitching function

        :return:
        """
        start = time.time()                   
        h5stitch                                            = self.h5f[self.stitch_loc]  
        self.updateParams(h5stitch.attrs)
        h5stitch.attrs['date']                              = str(datetime.datetime.now())
        h5stitch.attrs['creator']                           = u'matrix_overlaperr_stitch.py'
        
        self.h5msr                                          = self.h5f[self.msrEntry]
        instrScale                                          = self.h5msr['Instrument/scale_factor'][...] if 'Instrument/scale_factor' in self.h5msr else 1
        h5scans                                             = self.h5msr['Data']                                         
        stitch_step                                         = np.double(h5stitch['stitch_step'][...])  # used when motorX/Y not available
        self.pix_size                                       = np.double(self.h5msr['Instrument/resolution'][...])         #pixel resolution in mm or um
        self.saveMidResults                                 = True if ('save_intermediate_results' in h5stitch.keys() and h5stitch['save_intermediate_results'][...]=='Y') else False
        
        for i in h5scans.keys():
            if 'NX_class' in h5scans[i].attrs and h5scans[i].attrs['NX_class']=='NXdata': #loop over scans
                print('Stitching '+i)
                h5si                        = h5scans[i]
                h5fi                        = h5stitch.create_group('stitched_scans/'+h5si.attrs['name']) # create stitched folder with same name as data e.g.Scan_b1
                
#                 self.isSlopeStitch = True if 'slopes_x' in h5si else False
                #Apply external mask if exists
                mask                        = h5si['mask'][...]
                if 'input_mask' in h5stitch:
                    mask                    = applyExternalMask(mask, h5stitch['input_mask'][...])
                
                if not np.any(mask):
                    continue
                
                # do 1D stitching first
                mXArr                       = h5si['motor_X'][...]                                  #1D array
                if np.isnan(np.sum(mXArr)):                                                         #use stitch step if no data is available
                    mXArr                   = range(0,len(mXArr))*stitch_step
                else:
                    mXArr = -1.0*mXArr if np.all(np.diff(mXArr)<0) else mXArr
                    
                mXArr_rel                   = mXArr-min(mXArr)                                      #usually in mm
                mXArr_rp                    = np.rint(mXArr_rel/self.pix_size[0]).astype(int)          # pixel offsets from start (rp: relative & pixelwise)
                ########
                #TODO: interpolation of slp/hgt if step is not integer multiple of pixel
                szPat                       = mask.shape                                    # patch size XY, last dimension is longest i.e. stitching dir
                res_szX                     = szPat[-1]+max(mXArr_rp)                               # result size X stitch dirctn
                res_szY                     = szPat[-2]                                             # result size Y
                
                subRefFlag = True if 'ref' in h5si and (not self.h5msr.attrs['flag_ref_subtracted']) else False
                
                #stitching slopes
                if self.isSlopeStitch:
                    h5fi, C = self.stitchSlopes(h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale)
                #stitching height
                else:
                    h5fi, C = self.stitchHeights(h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale)
                    
                h5fi['intensity'] = []
                
                if self.saveMidResults:
                    h5fi['correctors'] = C
                    h5fi['correctors'].attrs[u'format'] = 'num of subapertures x [pitch,roll,piston]'
                addAttributesStitchedScans(h5si, h5fi, self.isSlopeStitch, subRefFlag, pix_size = self.pix_size, gravitySubFlag=False)# add attributes for scans

        self.avgScans(mXArr)
        print(time.time()-start)
        alertMsg('Finished','Finished stitching')

        # show()
    

[docs]    def stitchSlopes(self, h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale):
        """
        Slopes stitching function on i-th scan

        :param h5si: H5 scan object containing subaperture data of i-th scan
        :param h5fi: H5 stitched object
        :param subRefFlag: Subtract reference flag
        :param mask: Subaperture mask array
        :param mXArr_rp: Translation offset array in pixels
        :param res_szY: Stitched result size Y
        :param res_szX: Stitched result size X
        :param instrScale: Instrument scaling factor
        :return: H5 stitched object (h5fi), correctors
        """
        #####################################
        ## PreProcessing
        # subtract reference
        slxArr = h5si['slopes_x'][...] if not subRefFlag else h5si['slopes_x'][...]-h5si['ref/slopes_x'][...] #ref should be broadcasted to all patches
        slyArr = h5si['slopes_y'][...] if not subRefFlag else h5si['slopes_y'][...]-h5si['ref/slopes_y'][...]
        
        # reference from measured data
        if self.useRefMeasureData=='avg':
            slxArr,slyArr = preprocessSlp('REF_M_A',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
        elif self.useRefMeasureData=='avg_high_freq':
            slxArr,slyArr = preprocessSlp('REF_M_A_HF',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
        elif self.useRefMeasureData=='high_freq':
            slxArr,slyArr = preprocessSlp('REF_M_HF',slxArr,slyArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
            
        # correction for ellipse
        if self.useThreshold=='pre_process':
            slxArr,slyArr = preprocessSlp('E_FS',slxArr,slyArr, args=[self.thrMin, self.thrMax])
        #####################################
         
        if ~np.any(~np.isnan(slxArr)):
            return h5fi, []
        if ~np.any(~np.isnan(slyArr)):
            return h5fi, []
        Ax, Ex, validOvrX, validPatchesX, slxArr_i = self.buildOverlapErrors(mXArr_rp, slxArr, mask, res_szY, res_szX, otype='sx')
        Ay, Ey, validOvrY, validPatchesY, slyArr_i = self.buildOverlapErrors(mXArr_rp, slyArr, mask, res_szY, res_szX, otype='sy')
        
        minVLen = np.min([len(validOvrX), len(validOvrY)])
        validOvrX[:minVLen] = validOvrX[:minVLen] * validOvrY[:minVLen]
        validOvrY[:minVLen] = validOvrX[:minVLen] * validOvrY[:minVLen]
        validOvrX[minVLen:] = False
        validOvrY[minVLen:] = False
        validPat = validPatchesX*validPatchesY
        if ~np.any(validPat):
            return h5fi, []

        A = Ax[validOvrX]
        Ex = Ex[validOvrX]
        Ey = Ey[validOvrY]
        if self.oeType=='FULL':
            cx = self.getCorrectorsFullOE(A, Ex, validPat)
            cy = self.getCorrectorsFullOE(A, Ey, validPat)
            C = np.stack((cx, cy), axis=1)
        else:
            E = np.stack((Ex, Ey), axis=1) #Ex,Ey are 1D arays stacked to 2D array E
            invA = calcMatrixInv(A, validPat = validPat, method=self.invType)
            C = np.matmul(invA, E)
            
            if self.post_process == 'ref_cv':
                C[validPat,:] = signal.detrend(C[validPat,:], axis=0)
                
        h5fi['slopes_x'] = instrScale*self.correctAndJoin(C[:,0], slxArr_i, otype='sx', h5fi=h5fi) # scaling -0.5*1e3 is for sharper
        h5fi['slopes_y'] = instrScale*self.correctAndJoin(C[:,1], slyArr_i, otype='sy', h5fi=h5fi)
        
        return h5fi, C

    

[docs]    def stitchHeights(self, h5si, h5fi, subRefFlag, mask, mXArr_rp, res_szY, res_szX, instrScale):
        """
        Height stitching function on i-th scan

        :param h5si: H5 scan object containing subaperture data of i-th scan
        :param h5fi: H5 stitched object
        :param subRefFlag: Subtract reference flag
        :param mask: Subaperture mask array
        :param mXArr_rp: Translation offset array in pixels
        :param res_szY: Stitched result size Y
        :param res_szX: Stitched result size X
        :param instrScale: Instrument scaling factor
        :return: H5 stitched object (h5fi), correctors
        """
        #####################################
        ## PreProcessing
        # subtract reference
        hArr = h5si['height'][...] if not subRefFlag else h5si['height'][...]-h5si['ref/height'][...] #ref should be broadcasted to all patches
        # reference from measured data
        if self.useRefMeasureData=='avg':
            hArr = preprocessHgt('REF_M_A',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
        if self.useRefMeasureData=='avg_high_freq':
            hArr = preprocessHgt('REF_M_A_HF',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
        if self.useRefMeasureData=='high_freq':
            hArr = preprocessHgt('REF_M_HF',hArr, pix_size = self.pix_size, mask = mask, outH5Obj = h5fi, isSave = self.saveMidResults)
        #####################################

        print('Fin preprocessing')

        if ~np.any(~np.isnan(hArr)):
            return h5fi, []
        Ar, Er, validOvr, validPat, hArr_i = self.buildOverlapErrors(mXArr_rp, hArr, mask, res_szY, res_szX, otype='h')
        
        if ~np.any(validPat):
            return h5fi, []
        A = Ar[validOvr]
        E = Er[validOvr] # 3 cols pitch/roll/piston
        
        if self.oeType=='FULL':
            C = self.getCorrectorsFullOE(A, E, validPat)
            C = np.reshape(C, (3,len(validPat))).T
        else:
            invA = calcMatrixInv(A, validPat = validPat, method=self.invType)
            C = np.matmul(invA, E)
        print('Fin determining correctors array')

        if self.post_process == 'ref_cv':
            C[validPat,1:3] = signal.detrend(C[validPat,1:3], axis=0)
        
        # figure(101); plot(C[:,0]); title('Piston')
        # figure(103); plot(C[:,1]); title('Roll')
        # figure(103); plot(C[:,2]); title('Pitch')
        h5fi['height'] = instrScale*self.correctAndJoin(C, hArr_i, otype='h', h5fi=h5fi)
        print('Fin correction')

        return h5fi, C



[docs]    def getCorrectorsFullOE(self,A,E,validPat):
        """
        Get correctors when all pixels in the overlap are used

        :param A: IndexMatrix
        :param E: Overlap errors array
        :param validPat: Valid subapertures
        :return: Correctors array
        """
        xvRef, yvRef = getXYGrid(E, pix_size=self.pix_size, order=3, mask=~np.isnan(E))
        A = A[:,np.newaxis,np.newaxis,:]
        B = ~np.isnan(E)
        B = B[:,:,:,np.newaxis]
        A = A*B
        sa = A.shape
        A = np.reshape(A, (sa[0]*sa[1]*sa[2],sa[3]))
        E = np.reshape(E, (sa[0]*sa[1]*sa[2],1))
        valPix = ~np.isnan(E[:,0])
        A = A[valPix,:]
        xv= np.reshape(xvRef, (sa[0]*sa[1]*sa[2],1))[valPix]
        yv = np.reshape(yvRef, (sa[0]*sa[1]*sa[2],1))[valPix]

        if ~self.isSlopeStitch: # for height stitching c0+c1X+c2Y = e0
            A = np.concatenate((A,A*yv,A*xv),axis=1)
            if self.useRefMeasureData=='extract_from_stitching': # need to test
                A = self.buildAforExtractRef(A,xv,yv)
            A_shp = A.shape
            validPat = np.repeat(validPat, A_shp[1], axis=0)
        else:
            if self.useRefMeasureData=='extract_from_stitching': # need to test
                A = self.buildAforExtractRef(A,xv,yv)
        E = E[valPix,:]
        invA = calcMatrixInv(A, validPat=validPat, method=self.invType)
        C = np.matmul(invA, E)
        return C



[docs]    def buildAforExtractRef(self,A,xv,yv):
        """
        Expand index matrix to extract reference from measured data

        :param A: IndexMatrix
        :param xv: X-grid data in reference coordinates
        :param yv: Y-grid data in reference coordinates
        :return: Expanded index matrix
        """
        degree = self.degRef
        startDeg = self.startDegRef
        for n in range(startDeg, degree + 1):
            for k in range(0, n + 1):
                A = np.concatenate((A, A * xv ** (k) * yv ** (n - k)), axis=1)
        return A

[docs]    def buildOverlapErrors(self, oXArr, pat_arr, mask_arr, res_szY, res_szX, otype):
        """
        Overlap errors are retrieved as numpy ndarray

        :param oXArr: Translation offset array. Steps are in pixels
        :param pat_arr: Subaperture slope/height array
        :param mask_arr: Mask array with same shape as pat_arr
        :param res_szY: Stitched result data size Y
        :param res_szX: Stitched result data size X
        :param otype: Sata object type, 'sx' for slopes X, 'sy' for slopes Y, 'h' for height
        :return: IndexMatrix, overlap errors, valid overlaps, valid subapertures, subaperture data array
        """
        
        A, sarr_diff, Ep_diff, max_ov_cnt, validOverlaps, validPatches, sarr_ret, _,_ = buildOverlapMatrices(self, oXArr, pat_arr, mask_arr, res_szY, res_szX, otype, fitData= (self.oeType=='PF'), degRef=self.degRef, startDegRef=self.startDegRef, extractRef = (self.useRefMeasureData=='extract_from_stitching'))
        if self.oeType=='SM':
            # Method 1: overlap errors= simple average of 2D overlap diff image
            E = np.nanmean(np.nanmean(sarr_diff,axis=2),axis=1)
        elif self.oeType=='FULL':
            # Method 2: overlap errors=  2D overlap diff image
            E = sarr_diff
        else:#oetype=PF
            # Method 3: plane fit errors (default)
            E = Ep_diff if otype=='h' else Ep_diff[:,0]
                
        return A, E, validOverlaps, validPatches, sarr_ret

        """
        Apply a threshold on subaperture values

        :param carr: Corrected subapertures
        :param oarr: Original subapertures

[docs]    def correctAndJoin(self, c, sarr, otype, h5fi):
        """
        Apply corrections and join the subapertures

        :param c: Correctors
        :param sarr: Subaperture array
        :param otype: Data type - slope/height
        :param h5fi: H5 stitched object
        :return: Stitched image
        """
        if otype in ['sx', 'sy']:
            sarr_c = sarr - c.reshape(len(c),1,1)
            if self.useThreshold=='post_process':       sarr_c = self.maskByThreshold(sarr_c, sarr)
            sf = np.nanmean(sarr_c, axis=0)
            
            #figure; plot(sarr_c[:,int(sarr_c.shape[1]/2),:].T); title('Corrected patches centerline')
            return sf
        
        elif otype=='h':    
            patch_count,ny,nx           = sarr.shape
            xx,yy                       = getXYGrid(sarr[0,:,:], self.pix_size, order=2, mask=np.full(sarr[0,:,:].shape, True))
            
            if self.oeType=='SM':
                corPitch = 0
                corRoll = 0
                corPiston   = c.reshape(patch_count,1,1)
            else:
                corPitch    = c[:,2].reshape(patch_count,1,1)*xx
                corRoll     = c[:,1].reshape(patch_count,1,1)*yy
                corPiston   = c[:,0].reshape(patch_count,1,1)
            
            cor = 0
            if self.useRefMeasureData=='extract_from_stitching':    # polynomial fit only
                i = 0
                degree = self.degRef
                startDeg = self.startDegRef
                if self.oeType=='PF':
                    c = c[:,(startDeg+1)**2+1:] #ignore coeffs for degree less than startDeg
                for n in range(startDeg,degree+1):
                    for k in range(0,n+1):
                        cor = c[:,i].reshape(patch_count,1,1) if n==0 and k==0 else cor+(c[:,i].reshape(patch_count,1,1))*xx**(k)*yy**(n-k)
                        i+=1
            cor = cor + corPitch + corRoll + corPiston
            sarr_c = sarr - (cor)
            if self.saveMidResults:
                h5fi['correction_images'] = cor
            figure(11); plot(sarr_c[:,int(sarr_c.shape[1]/2),:].T); title('Corrected patches centerline')
            figure(13); plot(sarr[:,int(sarr.shape[1]/2),:].T); title(' subaps centerline')
            sf = np.nanmean(sarr_c, axis=0)
                
            return sf
