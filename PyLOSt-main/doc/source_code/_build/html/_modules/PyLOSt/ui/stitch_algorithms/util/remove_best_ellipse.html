[docs]def rotHght(x , z, idx):
    """
    Rotate height to center at array index idx

    :param x: Mirror positions along X
    :param z: Height data
    :param idx: Index
    :return: Rotated height
    """
    s = hgtToSlp1D(x, z)
    s = s - s[idx]
    zf = slpToHgt1D(x, s)
    zf = zf - zf[idx]
    return zf

    

[docs]def slpToHgt1D(x, d, intType='cumtrapz'):
    """
    Integrate slope to height (1D)

    :param x: Mirror positions along X
    :param d: slopes
    :param intType: Integration method e.g. cumtrapz
    :return:
    """
    if intType=='cumtrapz':
        return integrate.cumtrapz(d, x, initial=0)

    

[docs]def hgtToSlp1D(x, d, diffType='grad'):
    """
    Height to slope (1D)

    :param x: Mirror positions along X
    :param d: Height
    :param diffType: Derivative type e.g. gradient
    :return:
    """
    if diffType=='grad':
        return np.gradient(d,x)

[docs]def sfit_fn(xp, q, theta, tilt):
    """
    Slope fit function

    :param xp:
    :param q:
    :param theta:
    :param tilt:
    :return:
    """
    x = xp[0]
    p = xp[1]
    a = (p+q)/2; 
    b = np.sqrt(p*q)*np.sin(theta/1000);
    F = (1/2)*np.sqrt(p**2+q**2-2*p*q*np.cos(np.pi-(2*theta/1000)));
    alpha =np.arcsin(p/(2*F)*np.sin(np.pi-(2*theta/1000)));
    mu = alpha-(theta/1000);
    x0 = F - q*np.cos(alpha);
    y0 = -b*np.sqrt(1-(x0/a)**2);
    
    sx_fit_fn = (np.cos(mu)**2)*(b/a)*((np.cos(mu)*x+x0)/np.sqrt(-1*(np.cos(mu)**2)*(x**2)-2*np.cos(mu)*x*x0+(a**2)-(x0**2))) + np.cos(mu)*np.sin(-mu) + tilt
    sx_fit_fn = sx_fit_fn*1e6

    return sx_fit_fn



[docs]def zfit_fn(xp, q, theta, tilt):
    """
    Height fit function

    :param xp:
    :param q:
    :param theta:
    :param tilt:
    :return:
    """
    x = xp[0]
    p = xp[1]
    a = (p+q)/2; 
    b = np.sqrt(p*q)*np.sin(theta/1000);
    F = (1/2)*np.sqrt(p**2+q**2-2*p*q*np.cos(np.pi-(2*theta/1000)));
    alpha =np.arcsin(p/(2*F)*np.sin(np.pi-(2*theta/1000)));
    mu = alpha-(theta/1000);
    x0 = F - q*np.cos(alpha);
    y0 = -b*np.sqrt(1-(x0/a)**2);
    
    z1 = np.cos(mu)*(-np.sqrt((b**2)*(1-((x*np.cos(mu)+x0)/a)**2))-y0)
    z2 = np.sin(-mu)*x*np.cos(mu) 
    z3 =  np.cos(mu)*(y0+np.sqrt((b**2)*(1-(x0/a)**2)))
    z_fit_fn = z1+z2+z3 #+ tilt*x
    z_fit_fn = z_fit_fn*1e9
    
    return z_fit_fn
