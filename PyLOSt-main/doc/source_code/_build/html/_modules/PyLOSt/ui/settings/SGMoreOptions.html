[docs]    def setVals(self,options):
        if options[0]!='': self.leScans.setText(options[0])
        if options[1]!='': self.lePatches.setText(options[1])
        if options[2]!='': self.leSep.setText(options[2])
        if options[3]!='': self.leExt.setText(options[3])
        if options[4]!='':
            if 'F' in options[4]:
                self.isDirF.setChecked(True)
                self.leDirFstr.setText(options[4]['F'])
            else:
                self.isDirF.setChecked(False)
            if 'B' in options[4]:
                self.isDirB.setChecked(True)
                self.leDirBstr.setText(options[4]['B'])
            else:
                self.isDirB.setChecked(False)

        # if options[5]!='': self.leOrder.setText(options[5]) #TODO

[docs]    def getVals(self):
        dir = {}
        orderArr = [str(self.leOrder.item(i).whatsThis()) for i in range(self.leOrder.count())]
        if orderArr==self.orderArr0: orderArr=[]

        if self.isDirF.isChecked():
            dir['F'] = self.leDirFstr.text()
        if self.isDirB.isChecked():
            dir['B'] = self.leDirBstr.text()
        return [self.leScans.text(), self.lePatches.text(), self.leSep.text(), self.leExt.text(), dir, orderArr]
