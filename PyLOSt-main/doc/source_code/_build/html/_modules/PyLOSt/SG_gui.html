[docs]    def positionWindow(self, level=1, xsf=0):
        qtRectangle = self.frameGeometry()
        centerPoint = QDesktopWidget().availableGeometry().center()
        if level==2:
            centerPoint = centerPoint - QPoint(int(qtRectangle.width()/2)+10,0)
        if level==3 and xsf>0:
            centerPoint = QPoint(xsf,centerPoint.y()-50)
        qtRectangle.moveCenter(centerPoint)
        if qtRectangle.topLeft().x()>0 and qtRectangle.topLeft().y()>0:
            self.move(qtRectangle.topLeft())

[docs]    def setMoreOptText(self):
        mo = self.moreOpt
        txt = ''
        txt += 'Scans = ['+mo[0]+'];' if mo[0]!='' else ''
        txt += ' Subapertures = ['+mo[1]+'];' if mo[1]!='' else ''
        txt += ' Seperator = '+mo[2]+';' if mo[2]!='' else ''
        txt += ' Extension = '+mo[3]+';' if mo[3]!='' else ''
        txt += ' Directions = '+str(mo[4])+';' if any(mo[4]) else ''
        txt += ' Sequence = '+str(mo[5])+';' if any(mo[5]) else ''
        self.lbMore.setText(txt)
        self.lbMore.setToolTip(txt)
        self.lbMore.setStatusTip(txt)

[docs]    def loadInstrDataFolder(self):
        """Load folder with raw instrument data"""
        dg= QFileDialog()
        dg.setFileMode(QFileDialog.Directory)
        dg.setOption(QFileDialog.DontUseNativeDialog, True)
        dg.setOption(QFileDialog.ShowDirsOnly, False)
        dg.exec()
        self.loc_instr = dg.directory().absolutePath()
        # self.loc_instr = str(QFileDialog.getExistingDirectory(self, "Select Directory", self.loc_instr)) # old method
        self.rawInstrFolder.setText(self.loc_instr)
        fDir = os.path.split(os.path.splitext(self.loc_instr)[0])[1]
        self.msrName.setText(fDir)

        self.loadAdvancedFiles()
        
        # hide stats and header file
        self.label_conv_stats.hide()



[docs]    def loadAdvancedFiles(self):
        extension = Instruments.selectBy(id=self.curSelInstr)[0].dataFormats.split(',')
        for ext in extension:
            fl = [f for f in os.listdir(self.loc_instr) if f.endswith(ext.strip())]
            if any(fl):
                prefix = os.path.commonprefix(fl)
                self.lePrefix.setText(prefix)
                break

[docs]    def convH5(self):
        """Convert to h5 function. It redirects to a different function for each instrument"""
        try:
            if not self.validateForm():
                return

            self.progressBar.show()
            qin = Instruments.selectBy(id=self.curSelInstr)[0]
            qintype = InstrTypes.selectBy(id=qin.instrTypeID)[0]
            qinloc = Locations.selectBy(id=qin.instrLocationID)[0]
            # t = None
            # if qintype.instrType==commons.SHARPER:
            #     t=FuncThread(self.loc_instr, qin.instrId , qinloc.locationDesc, target=loadHasData, callback=self.finConvH5,callback_args=None)
            # if t: t.start()
            mName = self.msrName.text()
            if qintype.instrType == commons.SHARPER:
                loadHasData(self.loc_instr, qin.instrId, qinloc.locationDesc, self.loc_h5File, mName=mName, progress_bar = self.progressBar)
            elif qintype.instrType == commons.FIZEAU:
                loadFizData(self.loc_instr, qin.instrId, qinloc.locationDesc, self.loc_h5File, pf = self.lePrefix.text(), nbScans = int(self.leScans.text()), nbPatches = int(self.lePatches.text()), mName=mName, moreOpt=self.moreOpt, progress_bar = self.progressBar)
            elif qintype.instrType == commons.MSI:
                loadMsiData(self.loc_instr, qin.instrId, qinloc.locationDesc, self.loc_h5File, pf = self.lePrefix.text(), nbScans = int(self.leScans.text()), nbPatches = int(self.lePatches.text()), mName=mName, moreOpt=self.moreOpt, progress_bar = self.progressBar)
            elif qintype.instrType == commons.LTP:
                lp = self.lePatches.text()  
                nlp = int(lp) if lp!='' else 0
                loadSlp2Data(self.loc_instr, qin.instrId , qinloc.locationDesc, self.loc_h5File, pf = self.lePrefix.text(), nbScans = int(self.leScans.text()), nbPatches = nlp, mName=mName, moreOpt=self.moreOpt, progress_bar = self.progressBar)
            # self.overlay.show()
            self.progressBar.hide()
            alertMsg('Fin', 'Finished data format conversion')
            if os.path.exists(self.loc_h5File): self.open_h5file(self.loc_h5File)
        except Exception as e:
            print('convH5 <- SG_gui')
            print(e)

[docs]    def validateForm(self, valRawLoc=True):
        if self.msrName.text() == '':
            alertMsg('Mesurement entry', 'Please enter name of the measurement')
            return False
        if valRawLoc and self.loc_instr == '':
            alertMsg('Instrument data location', 'Please select the raw data location')
            return False
        if self.loc_h5File == '':
            alertMsg('save to', 'Please select h5 file to save')
            return False
        return True
